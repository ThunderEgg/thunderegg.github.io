67  FineIfaceInfo(std::shared_ptr<PatchInfo<D>> pinfo, Side<D> s)
68  {
69  nbr_info = pinfo->getFineNbrInfoPtr(s);
70  this->id = pinfo->id * Side<D>::num_sides + s.getIndex();
71  this->rank = pinfo->rank;
72  for (size_t i = 0; i < fine_ids.size(); i++) {

75  }
