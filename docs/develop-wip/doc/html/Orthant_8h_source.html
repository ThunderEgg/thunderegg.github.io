367  static std::array<Orthant, num_orthants / 2> getValuesOnSide(Side<D> s)
368  {
369  unsigned int bit_to_insert = s.getAxisIndex();
370  unsigned int set_bit = s.isLowerOnAxis() ? 0 : 1;
371  unsigned int lower_mask = ~((~0x0) << bit_to_insert);
372  unsigned int upper_mask = (~0x0) << (bit_to_insert + 1);
373 
374  std::array<Orthant<D>, Orthant<D>::num_orthants / 2> retval;
375  for (size_t i = 0; i < Orthant<D>::num_orthants / 2; i++) {
376  size_t value = (i << 1) & upper_mask;
377  value |= i & lower_mask;
378  value |= set_bit << bit_to_insert;
379  retval[i] = Orthant<D>(value);
380  }
381  return retval;
382  }
