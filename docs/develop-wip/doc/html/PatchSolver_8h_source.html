119  if (timer) {

125  }

135  {
136  if (timer) {
137  timer->start("Total Patch Smooth");
138  }
139  ghost_filler->fillGhost(u);
140  for (std::shared_ptr<const PatchInfo<D>> pinfo : domain->getPatchInfoVector()) {
141  if (timer) {
142  timer->start("Single Patch Solve");
143  }
144  solveSinglePatch(pinfo, u->getLocalData(pinfo->local_index),
145  f->getLocalData(pinfo->local_index));
146  if (timer) {

149  }
