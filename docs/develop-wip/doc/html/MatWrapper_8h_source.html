49  Vec getPetscVecWithoutGhost(std::shared_ptr<const Vector<D>> vec) const
50  {
51  Vec petsc_vec;
52  std::shared_ptr<const PETSc::VecWrapper<D>> petsc_vec_ptr
53  = std::dynamic_pointer_cast<const PETSc::VecWrapper<D>>(vec);
54  if (petsc_vec_ptr != nullptr && petsc_vec_ptr->getNumGhostCells() == 0) {
55  petsc_vec = petsc_vec_ptr->getVec();
56  } else {
57  // have to create a new petsc vector without ghostcells for petsc call
58  VecCreateMPI(vec->getMPIComm(), vec->getNumLocalCells(), PETSC_DETERMINE, &petsc_vec);
59  }
60  return petsc_vec;
61  }
68  void copyToPetscVec(std::shared_ptr<const Vector<D>> vec, Vec petsc_vec) const
69  {
70  std::shared_ptr<const PETSc::VecWrapper<D>> petsc_vec_ptr
71  = std::dynamic_pointer_cast<const PETSc::VecWrapper<D>>(vec);
72  if (petsc_vec_ptr == nullptr || petsc_vec_ptr->getNumGhostCells() > 0) {
73  double *petsc_vec_view;
74  size_t curr_index = 0;
75  VecGetArray(petsc_vec, &petsc_vec_view);
76  for (int i = 0; i < vec->getNumLocalPatches(); i++) {
77  const LocalData<D> ld = vec->getLocalData(i);
78  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
79  petsc_vec_view[curr_index] = ld[coord];
80  curr_index++;
81  });
82  }
83  VecRestoreArray(petsc_vec, &petsc_vec_view);
84  }
85  }
92  void copyToVec(Vec petsc_vec, std::shared_ptr<Vector<D>> vec) const
93  {
94  std::shared_ptr<PETSc::VecWrapper<D>> petsc_vec_ptr
95  = std::dynamic_pointer_cast<PETSc::VecWrapper<D>>(vec);
96  if (petsc_vec_ptr == nullptr || petsc_vec_ptr->getNumGhostCells() > 0) {
97  const double *petsc_vec_view;
98  size_t curr_index = 0;
99  VecGetArrayRead(petsc_vec, &petsc_vec_view);
100  for (int i = 0; i < vec->getNumLocalPatches(); i++) {
101  LocalData<D> ld = vec->getLocalData(i);
102  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
103  ld[coord] = petsc_vec_view[curr_index];
104  curr_index++;
105  });
106  }
107  VecRestoreArrayRead(petsc_vec, &petsc_vec_view);
108  }
109  }
116  void destroyPetscVec(std::shared_ptr<const Vector<D>> vec, Vec petsc_vec) const
117  {
118  std::shared_ptr<const PETSc::VecWrapper<D>> petsc_vec_ptr
119  = std::dynamic_pointer_cast<const PETSc::VecWrapper<D>>(vec);
120  if (petsc_vec_ptr == nullptr || petsc_vec_ptr->getNumGhostCells() > 0) {
121  VecDestroy(&petsc_vec);
122  }
123  }
