46  void visit(const Level<D> &level, std::list<std::shared_ptr<Vector<D>>> &u_vectors,
47  std::list<std::shared_ptr<const Vector<D>>> &f_vectors) const
48  {
49  if (level.coarsest()) {
50  for (int i = 0; i < num_coarse_sweeps; i++) {
51  this->smooth(level, u_vectors, f_vectors);
52  }
53  } else {
54  for (int i = 0; i < num_pre_sweeps; i++) {
55  this->smooth(level, u_vectors, f_vectors);
56  }
57  this->prepCoarser(level, u_vectors, f_vectors);
58  this->visit(*level.getCoarser(), u_vectors, f_vectors);
59  for (int i = 0; i < num_post_sweeps; i++) {
60  this->smooth(level, u_vectors, f_vectors);
61  }
62  }
63  if (!level.finest()) {
64  this->prepFiner(level, u_vectors, f_vectors);
65  }
66  }
