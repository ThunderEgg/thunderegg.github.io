78  static void
79  IndexIfacesLocal(const std::map<int, std::shared_ptr<Interface<D>>> & id_to_iface_map,
80  const std::vector<std::shared_ptr<PatchIfaceInfo<D>>> &piinfos,
81  std::vector<std::shared_ptr<Interface<D>>> & interfaces)
82  {
83  int curr_local_index = 0;
84 
85  // index interface objects first
86  interfaces.clear();
87  interfaces.reserve(id_to_iface_map.size());
88  for (auto pair : id_to_iface_map) {
89  auto iface = pair.second;
90 
91  iface->local_index = curr_local_index;
92  interfaces.push_back(iface);
93 
94  for (auto patch : iface->patches) {
95  if (patch.type.isNormal() || patch.type.isFineToFine()
96  || patch.type.isCoarseToCoarse()) {
97  patch.getNonConstPiinfo()->getIfaceInfo(patch.side)->patch_local_index
98  = curr_local_index;
99  patch.getNonConstPiinfo()->getIfaceInfo(patch.side)->col_local_index
100  = curr_local_index;
101  patch.getNonConstPiinfo()->getIfaceInfo(patch.side)->row_local_index
102  = curr_local_index;
103  } else if (patch.type.isFineToCoarse()) {
104  patch.getNonConstPiinfo()
105  ->getCoarseIfaceInfo(patch.side)
106  ->coarse_col_local_index
107  = curr_local_index;
108  } else if (patch.type.isCoarseToFine()) {
109  auto iface_info = patch.getNonConstPiinfo()->getFineIfaceInfo(patch.side);
110  for (size_t i = 0; i < iface_info->fine_col_local_indexes.size(); i++) {
111  if (iface_info->fine_ids[i] == iface->id) {
112  iface_info->fine_col_local_indexes[i] = curr_local_index;
113  break;
114  }
115  }
116  }
117  }
118 
119  curr_local_index++;
120  }
121 
122  // perform rest of necessary indexing
123  IndexRemainingColIfacesLocal(curr_local_index, interfaces);
124  IndexRemainingRowIfacesLocal(curr_local_index, interfaces);
125  IndexRemainingPatchIfacesLocal(curr_local_index, piinfos);
126  }

255  static void IndexIfacesGlobal(
256  const std::map<int, std::map<int, std::shared_ptr<Schur::Interface<D>>>> &rank_id_iface_map,
257  const std::vector<std::shared_ptr<Interface<D>>> & interfaces,
258  const std::vector<std::shared_ptr<PatchIfaceInfo<D>>> & piinfos)
259  {
260  // get starting global index for this rank
261  int starting_global_index;
262  int num_local_interfaces = (int) interfaces.size();
263  MPI_Scan(&num_local_interfaces, &starting_global_index, 1, MPI_INT, MPI_SUM,
264  MPI_COMM_WORLD);
265  starting_global_index -= num_local_interfaces;
266 
267  // index local interfaces first
268  for (auto iface : interfaces) {
269  iface->global_index = starting_global_index + iface->local_index;
270 
271  for (auto patch : iface->patches) {
272  if (patch.type.isNormal() || patch.type.isFineToFine()
273  || patch.type.isCoarseToCoarse()) {
274  auto iface_info = patch.getNonConstPiinfo()->getIfaceInfo(patch.side);
275  iface_info->global_index = iface->global_index;
276  } else if (patch.type.isFineToCoarse()) {
277  auto iface_info = patch.getNonConstPiinfo()->getCoarseIfaceInfo(patch.side);
278  iface_info->coarse_global_index = iface->global_index;
279  } else if (patch.type.isCoarseToFine()) {
280  auto iface_info = patch.getNonConstPiinfo()->getFineIfaceInfo(patch.side);
281  for (size_t i = 0; i < iface_info->fine_col_local_indexes.size(); i++) {
282  if (iface_info->fine_ids[i] == iface->id) {
283  iface_info->fine_global_indexes[i] = iface->global_index;
284  break;
285  }
286  }
287  }
288  }
289  }
290  SendAndReceiveGlobalIndexes(rank_id_iface_map, interfaces, piinfos);
291  }
292  static void SendAndReceiveGlobalIndexes(
293  const std::map<int, std::map<int, std::shared_ptr<Schur::Interface<D>>>> &rank_id_iface_map,
294  const std::vector<std::shared_ptr<Interface<D>>> & interfaces,
295  const std::vector<std::shared_ptr<PatchIfaceInfo<D>>> & piinfos)
296  {
297  std::map<int, std::map<int, std::set<int *>>> rank_to_id_to_global_indexes_to_set;
298  std::deque<std::vector<int>> recv_buffers;
299  std::vector<MPI_Request> recv_requests;
300  SetupGlobalIndexRecvRequests(interfaces, piinfos, rank_to_id_to_global_indexes_to_set,
301  recv_buffers, recv_requests);
302 
303  std::deque<std::vector<int>> send_buffers;
304  std::vector<MPI_Request> send_requests;
305  SetupGlobalIndexSendRequests(rank_id_iface_map, interfaces, piinfos, send_buffers,
306  send_requests);
307 
308  size_t num_recvs = recv_requests.size();
309  for (size_t i = 0; i < num_recvs; i++) {
310  int index;
311  MPI_Status status;
312  MPI_Waitany(recv_requests.size(), recv_requests.data(), &index, &status);
313  const std::vector<int> &buffer = recv_buffers[index];
314  const auto & id_to_global_indexes_to_set
315  = rank_to_id_to_global_indexes_to_set[status.MPI_SOURCE];
316 
317  // set the global indexes
318  size_t curr_index = 0;
319  for (auto pair : id_to_global_indexes_to_set) {
320  for (int *global_index_to_set : pair.second) {
321  *global_index_to_set = buffer[curr_index];
322  }
323  curr_index++;
324  }
325  }
326 
327  MPI_Waitall((int) send_requests.size(), send_requests.data(), MPI_STATUSES_IGNORE);
328  }

463

465  send_buffers.emplace_back();

467  send_buffer.reserve(pair.second.size());
468

470  send_buffer.push_back(id_global_index_pair.second);

473  MPI_Request request;
474  MPI_Isend(send_buffer.data(), (int) send_buffer.size(), MPI_INT, pair.first, 0,
475  MPI_COMM_WORLD, &request);
476  send_requests.push_back(request);

508  rank_to_id_and_global_index_pairs[iface_info->rank].emplace(
509  iface->id, iface->global_index);
510  }
511  }

516  // this interface will affect the values of the outer interfaces

560  int num_global_ifaces = 0;
561  int iface_stride;
562  std::array<int, D - 1> lengths;

564  public:
565  InterfaceDomain() = default;
572  explicit InterfaceDomain(std::shared_ptr<const Domain<D>> domain)
573  {
574  this->domain = domain;
575  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
576  iface_stride = 1;
577  for (size_t i = 0; i < D - 1; i++) {
578  iface_stride *= domain->getNs()[i];
579  lengths[i] = domain->getNs()[i];
580  }
581  std::vector<std::shared_ptr<PatchIfaceInfo<D>>> piinfos_non_const;
582  piinfos.reserve(domain->getNumLocalPatches());
583  piinfos_non_const.reserve(domain->getNumLocalPatches());
584  for (auto &pinfo : domain->getPatchInfoVector()) {
585  piinfos_non_const.emplace_back(new PatchIfaceInfo<D>(pinfo));
586  piinfos.push_back(piinfos_non_const.back());
587  }
588  std::map<int, std::map<int, std::shared_ptr<Schur::Interface<D>>>> rank_id_iface_map;
589  std::vector<std::shared_ptr<Schur::PatchIfaceInfo<D>>> off_proc_piinfos;
590  Interface<D>::EnumerateIfacesFromPiinfoVector(piinfos, rank_id_iface_map, off_proc_piinfos);
591 
592  std::vector<std::shared_ptr<Schur::Interface<D>>> interfaces_non_const;
593  IndexIfacesLocal(rank_id_iface_map[rank], piinfos_non_const, interfaces_non_const);
594  IndexIfacesGlobal(rank_id_iface_map, interfaces_non_const, piinfos_non_const);
595 
596  interfaces.reserve(interfaces_non_const.size());
597  for (auto iface : interfaces_non_const) {
598  interfaces.push_back(iface);
599  }
600 
601  int num_ifaces = interfaces.size();
602  MPI_Allreduce(&num_ifaces, &num_global_ifaces, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
603  }

605  void updateInterfaceDist(std::shared_ptr<Vector<D - 1>> gamma)
606  {
607  // send outgoing messages
608  std::vector<MPI_Request> requests;
609  requests.reserve(patch_in_id_local_rank_vec.size() + patch_out_id_local_rank_vec.size());
610 
611  // send info
612  for (const auto &tuple : patch_out_id_local_rank_vec) {
613  int id = std::get<0>(tuple);
614  int local_index = std::get<1>(tuple);
615  int dest = std::get<2>(tuple);
616  MPI_Request request;
617  MPI_Isend(gamma->getLocalData(local_index).getPtr(), iface_stride, MPI_DOUBLE, dest, id,
618  MPI_COMM_WORLD, &request);
619  requests.push_back(request);
620  }
621  // recv info
622  for (const auto &tuple : patch_in_id_local_rank_vec) {
623  int id = std::get<0>(tuple);
624  int local_index = std::get<1>(tuple);
625  int source = std::get<2>(tuple);
626 
627  double buffer[iface_stride];
628  MPI_Recv(buffer, iface_stride, MPI_DOUBLE, source, id, MPI_COMM_WORLD,
