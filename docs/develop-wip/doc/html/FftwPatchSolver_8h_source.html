111 template <size_t D> void FftwPatchSolver<D>::addPatch(ThunderEgg::Schur::SchurInfo<D> &sinfo)
112 {
113  using namespace std;
114  if (!initialized) {
115  initialized = true;
116  for (size_t i = 0; i <= D; i++) {
117  npow[i] = (int) std::pow(n, i);
118  }
119  std::array<int, D> lengths;
120  lengths.fill(n);
121  f_copy = std::make_shared<ValVector<D>>(MPI_COMM_SELF, lengths, 0, 1);
122  tmp = std::make_shared<ValVector<D>>(MPI_COMM_SELF, lengths, 0, 1);
123  sol = std::make_shared<ValVector<D>>(MPI_COMM_SELF, lengths, 0, 1);
124  }
125 
126  int ns[D];
127  fftw_r2r_kind transforms[D];
128  fftw_r2r_kind transforms_inv[D];
129  if (!plan1.count(sinfo)) {
130  for (size_t i = 0; i < D; i++) {
131  ns[D - 1 - i] = n;
132  // x direction
133  if (sinfo.pinfo->isNeumann(Side<D>(2 * i))
134  && sinfo.pinfo->isNeumann(Side<D>(2 * i + 1))) {
135  transforms[D - 1 - i] = FFTW_REDFT10;
136  transforms_inv[D - 1 - i] = FFTW_REDFT01;
137  } else if (sinfo.pinfo->isNeumann(Side<D>(2 * i))) {
138  transforms[D - 1 - i] = FFTW_REDFT11;
139  transforms_inv[D - 1 - i] = FFTW_REDFT11;
140  } else if (sinfo.pinfo->isNeumann(Side<D>(2 * i + 1))) {
141  transforms[D - 1 - i] = FFTW_RODFT11;
142  transforms_inv[D - 1 - i] = FFTW_RODFT11;
143  } else {
144  transforms[D - 1 - i] = FFTW_RODFT10;
145  transforms_inv[D - 1 - i] = FFTW_RODFT01;
146  }
147  }
148 
149  plan1[sinfo] = fftw_plan_r2r(D, ns, &f_copy->vec[0], &tmp->vec[0], transforms,
150  FFTW_MEASURE | FFTW_DESTROY_INPUT);
151  plan2[sinfo] = fftw_plan_r2r(D, ns, &tmp->vec[0], &sol->vec[0], transforms_inv,
152  FFTW_MEASURE | FFTW_DESTROY_INPUT);
153  }
154 
155  if (!denoms.count(sinfo)) {
156  valarray<double> &denom = denoms[sinfo];
157  denom.resize(pow(n, D));
158 
159  valarray<double> ones(pow(n, D - 1));
160  ones = 1;
161 
162  for (size_t i = 0; i < D; i++) {
163  valarray<size_t> sizes(D - 1);
164  sizes = n;
165  valarray<size_t> strides(D - 1);
166  for (size_t sinfo = 1; sinfo < D; sinfo++) {
167  strides[sinfo - 1] = pow(n, (i + sinfo) % D);
168  }
169  double h = sinfo.pinfo->spacings[i];
170 
171  if (sinfo.pinfo->isNeumann(Side<D>(i * 2))
172  && sinfo.pinfo->isNeumann(Side<D>(i * 2 + 1))) {
173  for (int xi = 0; xi < n; xi++) {
174  denom[gslice(xi * pow(n, i), sizes, strides)]
175  -= 4 / (h * h) * pow(sin(xi * M_PI / (2 * n)), 2) * ones;
176  }
177  } else if (sinfo.pinfo->isNeumann(Side<D>(i * 2))
178  || sinfo.pinfo->isNeumann(Side<D>(i * 2 + 1))) {
179  for (int xi = 0; xi < n; xi++) {
180  denom[gslice(xi * pow(n, i), sizes, strides)]
181  -= 4 / (h * h) * pow(sin((xi + 0.5) * M_PI / (2 * n)), 2) * ones;
182  }
183  } else {
184  for (int xi = 0; xi < n; xi++) {
185  denom[gslice(xi * pow(n, i), sizes, strides)]
186  -= 4 / (h * h) * pow(sin((xi + 1) * M_PI / (2 * n)), 2) * ones;
187  }
188  }
189  }
190 
191  denom += lambda;
192  }
193 }
194 template <size_t D>
195 void FftwPatchSolver<D>::solve(ThunderEgg::Schur::SchurInfo<D> &sinfo,
196  std::shared_ptr<const Vector<D>> f, std::shared_ptr<Vector<D>> u,
197  std::shared_ptr<const Vector<D - 1>> gamma)
198 {
199  using namespace std;
200  const LocalData<D> f_view = f->getLocalData(sinfo.pinfo->local_index);
201  LocalData<D> f_copy_view = f_copy->getLocalData(0);
202  LocalData<D> tmp_view = tmp->getLocalData(0);
203 
204  std::array<int, D> start, end;
205  start.fill(0);
206  end.fill(n - 1);
207 
208  nested_loop<D>(start, end,
209  [&](std::array<int, D> coord) { f_copy_view[coord] = f_view[coord]; });
210 
211  StarPatchOperator<D> op(sh);
212  op.addInterfaceToRHS(sinfo, gamma, f_copy->getLocalData(0));
213 
214  fftw_execute(plan1[sinfo]);
215 
216  tmp->vec /= denoms[sinfo];
217 
218  if (sinfo.pinfo->neumann.all()) {
219  tmp->vec[0] = 0;
220  }
221 
222  fftw_execute(plan2[sinfo]);
223 
224  sol->vec /= pow(2.0 * n, D);
225 
226  LocalData<D> u_view = u->getLocalData(sinfo.pinfo->local_index);
227  LocalData<D> sol_view = sol->getLocalData(0);
228  nested_loop<D>(start, end, [&](std::array<int, D> coord) { u_view[coord] = sol_view[coord]; });
229 }
