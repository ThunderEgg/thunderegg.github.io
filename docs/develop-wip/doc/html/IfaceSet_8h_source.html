102  void insert(int id, Side<D> s, std::shared_ptr<PatchIfaceInfo<D>> sinfo)
103  {
104  this->id = id;
105  sides.push_back(s);
106  sinfos.push_back(sinfo);
107  switch (sinfo->pinfo->getNbrType(s)) {
108  case NbrType::Normal:
109  types.push_back(IfaceType<D>::normal);
110  break;
111  case NbrType::Fine: {
112  auto info = sinfo->getFineIfaceInfo(s);
113  if (info->id == id) {
114  types.push_back(IfaceType<D>::coarse_to_coarse);
115  } else {
116  for (Orthant<D - 1> o : Orthant<D - 1>::getValues()) {
117  if (info->fine_ids[o.getIndex()] == id) {
118  types.push_back(IfaceType<D>::coarse_to_fine);
119  types.back().setOrthant(o);
120  break;
121  }
122  }
123  }
124  } break;
125  case NbrType::Coarse: {
126  auto info = sinfo->getCoarseIfaceInfo(s);
127  if (info->id == id) {
128  types.push_back(IfaceType<D>::fine_to_fine);
129  } else {
130  types.push_back(IfaceType<D>::fine_to_coarse);
131  types.back().setOrthant(info->orth_on_coarse);
132  }
133  } break;
134  }
135  }

209 std::map<int, IfaceSet<D>> IfaceSet<D>::EnumerateIfaces(Iter begin, Iter end)
210 {
211  int rank;
212  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
213  std::map<int, IfaceSet<D>> ifaces;
214  std::set<int> incoming_procs;
215  std::map<int, std::map<int, IfaceSet<D>>> off_proc_ifaces;
216  for (; begin != end; begin++) {
217  std::shared_ptr<PatchIfaceInfo<D>> sinfo = *begin;
218  for (Side<D> s : Side<D>::getValues()) {
219  if (sinfo->pinfo->hasNbr(s)) {
220  switch (sinfo->pinfo->getNbrType(s)) {
221  case NbrType::Normal: {
222  auto info = sinfo->getNormalIfaceInfo(s);
223  if (info->rank == rank) {
224  ifaces[info->id].insert(info->id, s, sinfo);
225  if (info->nbr_info->rank != rank) {
226  incoming_procs.insert(info->nbr_info->rank);
227  }
228  } else {
229  off_proc_ifaces[info->rank][info->id].insert(info->id, s, sinfo);
230  }
231  } break;
232  case NbrType::Fine: {
233  auto info = sinfo->getFineIfaceInfo(s);
234  ifaces[info->id].insert(info->id, s, sinfo);
235  for (size_t i = 0; i < Orthant<D - 1>::num_orthants; i++) {
236  if (info->fine_ranks[i] == rank) {
237  ifaces[info->fine_ids[i]].insert(info->fine_ids[i], s, sinfo);
238  } else {
239  off_proc_ifaces[info->fine_ranks[i]][info->fine_ids[i]].insert(
240  info->fine_ids[i], s, sinfo);
241  incoming_procs.insert(info->fine_ranks[i]);
242  }
243  }
244  } break;
245  case NbrType::Coarse: {
246  auto info = sinfo->getCoarseIfaceInfo(s);
247  ifaces[info->id].insert(info->id, s, sinfo);
248  if (info->coarse_rank == rank) {
249  ifaces[info->coarse_id].insert(info->coarse_id, s, sinfo);
250  } else {
251  off_proc_ifaces[info->coarse_rank][info->coarse_id].insert(
252  info->coarse_id, s, sinfo);
253  incoming_procs.insert(info->coarse_rank);
254  }
255  } break;
256  }
257  }
258  }
259  }
260  // send info
261  std::deque<char *> buffers;
262  std::vector<MPI_Request> send_requests;
263  for (auto &p : off_proc_ifaces) {
264  int dest = p.first;
265  int size = 0;
266  for (auto q : p.second) {
267  IfaceSet<D> &iface = q.second;
268  size += iface.serialize(nullptr);
269  }
270  char *buffer = new char[size];
271  buffers.push_back(buffer);
272  int pos = 0;
273  for (auto q : p.second) {
274  IfaceSet<D> &iface = q.second;
275  pos += iface.serialize(buffer + pos);
276  }
277  MPI_Request request;
278  MPI_Isend(buffer, size, MPI_BYTE, dest, 0, MPI_COMM_WORLD, &request);
279  send_requests.push_back(request);
280  }
281  // recv info
282  for (int src : incoming_procs) {
283  MPI_Status status;
284  MPI_Probe(src, 0, MPI_COMM_WORLD, &status);
285  int size;
286  MPI_Get_count(&status, MPI_BYTE, &size);
287  char *buffer = new char[size];
288 
289  MPI_Recv(buffer, size, MPI_BYTE, src, 0, MPI_COMM_WORLD, &status);
290 
291  BufferReader reader(buffer);
292  while (reader.getPos() < size) {
293  IfaceSet<D> ifs;
294  reader >> ifs;
295  ifaces[ifs.id].merge(ifs);
296  }
297 
298  delete[] buffer;
299  }
300  // wait for all
301  MPI_Waitall(send_requests.size(), &send_requests[0], MPI_STATUSES_IGNORE);
302  // delete send buffers
303  for (char *buffer : buffers) {
304  delete[] buffer;
305  }
306  return ifaces;
307 }
