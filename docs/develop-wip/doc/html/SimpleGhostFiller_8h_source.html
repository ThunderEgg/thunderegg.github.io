50  void fillGhost(std::shared_ptr<const Vector<D>> u) const
51  {
52  for (auto pinfo : domain->getPatchInfoVector()) {
53  for (Side<D> s : Side<D>::getValues()) {
54  if (pinfo->hasNbr(s)) {
55  auto nbr_info = pinfo->getNormalNbrInfo(s);
56  const LocalData<D> this_patch = u->getLocalData(pinfo->local_index);
57  const LocalData<D> other_patch = u->getLocalData(nbr_info.local_index);
58  auto this_side = this_patch.getSliceOnSide(s);
59  auto other_side = other_patch.getGhostSliceOnSide(s.opposite(), 1);
60  nested_loop<D - 1>(this_side.getStart(), this_side.getEnd(),
61  [&](const std::array<int, D - 1> &coord) {
62  other_side[coord] = this_side[coord];
63  });
64  }
65  }
66  }
