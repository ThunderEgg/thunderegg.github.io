115  LocalData<D> getLocalDataForBuffer(double *buffer_ptr, const Side<D> side) const
116  {
117  auto ns = domain->getNs();
118  int num_ghost_cells = domain->getNumGhostCells();
119  // determine striding
120  std::array<int, D> strides;
121  strides[0] = 1;
122  for (size_t i = 1; i < D; i++) {
123  if (i == side.getAxisIndex() + 1) {
124  strides[i] = num_ghost_cells * strides[i - 1];
125  } else {
126  strides[i] = ns[i - 1] * strides[i - 1];
127  }
128  }
129  // transform buffer ptr so that it points to first non-ghost cell
130  double *transformed_buffer_ptr;
131  if (side.isLowerOnAxis()) {
132  transformed_buffer_ptr = buffer_ptr - (-num_ghost_cells) * strides[side.getAxisIndex()];
133  } else {
134  transformed_buffer_ptr
135  = buffer_ptr - ns[side.getAxisIndex()] * strides[side.getAxisIndex()];
136  }
137 
138  LocalData<D> buffer_data(transformed_buffer_ptr, strides, ns, num_ghost_cells);
139  return buffer_data;
140  }

159  MPIGhostFiller(std::shared_ptr<const Domain<D>> domain_in, int side_cases_in)
160  : domain(domain_in), side_cases(side_cases_in)
161  {
162  int rank;
163  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
164  // rank nbr_id side orthant nbrtype pinfo local_index
165  std::set<std::tuple<int, int, const Side<D>, const Orthant<D>, const NbrType,
166  std::shared_ptr<const PatchInfo<D>>, int>>
167  remote_call_set;
168  // rank id side local index
169  std::set<std::tuple<int, int, const Side<D>, int>> incoming_ghost_set;
170 
171  std::array<size_t, D> axis_ghost_lengths;
172  for (size_t axis = 0; axis < D; axis++) {
173  size_t length = 1;
174  for (size_t i = 0; i < D; i++) {
175  if (i == axis) {
176  length *= domain->getNumGhostCells();
177  } else {
178  length *= domain->getNs()[i];
179  }
180  }
181  axis_ghost_lengths[axis] = length;
182  }
183  std::set<int> ranks;
184  for (auto pinfo : domain->getPatchInfoVector()) {
185  for (Side<D> s : Side<D>::getValues()) {
186  if (pinfo->hasNbr(s)) {
187  switch (pinfo->getNbrType(s)) {
188  case NbrType::Normal: {
189  auto nbrinfo = pinfo->getNormalNbrInfo(s);
190  if (nbrinfo.rank == rank) {
191  local_calls.emplace_back(pinfo, s, NbrType::Normal,
192  Orthant<D>::null(), pinfo->local_index,
193  nbrinfo.local_index);
194  } else {
195  ranks.insert(nbrinfo.rank);
196  remote_call_set.emplace(nbrinfo.rank, nbrinfo.id, s.opposite(),
197  Orthant<D>::null(), NbrType::Normal, pinfo,
198  pinfo->local_index);
199  incoming_ghost_set.emplace(nbrinfo.rank, pinfo->id, s,
200  pinfo->local_index);
201  }
202  } break;
203  case NbrType::Fine: {
204  auto nbrinfo = pinfo->getFineNbrInfo(s);
205  auto orthants = Orthant<D>::getValuesOnSide(s);
206  for (size_t i = 0; i < orthants.size(); i++) {
207  if (nbrinfo.ranks[i] == rank) {
208  local_calls.emplace_back(pinfo, s, NbrType::Fine, orthants[i],
209  pinfo->local_index,
210  nbrinfo.local_indexes[i]);
211  } else {
212  ranks.insert(nbrinfo.ranks[i]);
213  remote_call_set.emplace(
214  nbrinfo.ranks[i], nbrinfo.ids[i], s.opposite(), orthants[i],
215  NbrType::Fine, pinfo, pinfo->local_index);
216  incoming_ghost_set.emplace(nbrinfo.ranks[i], pinfo->id, s,
217  pinfo->local_index);
218  }
219  }
220  } break;
221  case NbrType::Coarse: {
222  auto nbrinfo = pinfo->getCoarseNbrInfo(s);
223  auto orthant = Orthant<D>::getValuesOnSide(
224  s.opposite())[nbrinfo.orth_on_coarse.getIndex()];
225  if (nbrinfo.rank == rank) {
226  local_calls.emplace_back(pinfo, s, NbrType::Coarse, orthant,
227  pinfo->local_index, nbrinfo.local_index);
228  } else {
229  ranks.insert(nbrinfo.rank);
230  remote_call_set.emplace(nbrinfo.rank, nbrinfo.id, s.opposite(),
231  orthant, NbrType::Coarse, pinfo,
232  pinfo->local_index);
233  incoming_ghost_set.emplace(nbrinfo.rank, pinfo->id, s,
234  pinfo->local_index);
235  }
236  } break;
237  }
238  }
239  }
240  }
241  std::map<int, size_t> rank_index_map;
242  index_rank_map.reserve(ranks.size());
243  int curr_index = 0;
244  for (int rank : ranks) {
245  rank_index_map[rank] = curr_index;
246  index_rank_map.push_back(rank);
247  curr_index++;
248  }
249  std::tuple<int, Side<D>> prev_id_side;
250  send_buff_lengths.resize(ranks.size());
251  remote_calls.resize(ranks.size());
252  for (auto call : remote_call_set) {
253  int local_buffer_index = rank_index_map[std::get<0>(call)];
254  int id = std::get<1>(call);
255  auto side = std::get<2>(call).opposite();
256  auto orthant = std::get<3>(call);
257  auto nbr_type = std::get<4>(call);
258  auto pinfo = std::get<5>(call);
259  auto local_index = std::get<6>(call);
260 
261  size_t offset = send_buff_lengths[local_buffer_index];
262 
263  // calculate length in buffer need for ghost cells
264  size_t length = axis_ghost_lengths[side.getAxisIndex()];
265  // add length to buffer length
266  // if its the same side of the patch, resuse the previous buffer space
267  if (std::make_tuple(id, side) == prev_id_side) {
268  offset -= length;
269  } else {
270  send_buff_lengths[local_buffer_index] += length;
271  }
272 
273  remote_calls[local_buffer_index].emplace_back(pinfo, side, nbr_type, orthant,
274  local_index, offset);
275  prev_id_side = std::make_tuple(id, side);
276  }
277  recv_buff_lengths.resize(ranks.size());
278  incoming_ghosts.resize(ranks.size());
279  for (auto t : incoming_ghost_set) {
280  int local_buffer_index = rank_index_map[std::get<0>(t)];
281  Side<D> side = std::get<2>(t);
282  int local_index = std::get<3>(t);
283 
284  // add length for ghosts to buffer length
285  size_t length = axis_ghost_lengths[side.getAxisIndex()];
286  size_t offset = recv_buff_lengths[local_buffer_index];
287  recv_buff_lengths[local_buffer_index] += length;
288 
289  // add ghost to incoming ghosts
290  incoming_ghosts[local_buffer_index].emplace_back(local_index, side, offset);
291  }
292  }
298  void fillGhost(std::shared_ptr<const Vector<D>> u) const
299  {
300  // zero out ghost cells
301  for (auto pinfo : domain->getPatchInfoVector()) {
302  const LocalData<D> this_patch = u->getLocalData(pinfo->local_index);
303  for (Side<D> s : Side<D>::getValues()) {
304  if (pinfo->hasNbr(s)) {
305  for (int i = 0; i < pinfo->num_ghost_cells; i++) {
306  auto this_ghost = this_patch.getGhostSliceOnSide(s, i + 1);
307  nested_loop<D - 1>(
308  this_ghost.getStart(), this_ghost.getEnd(),
309  [&](const std::array<int, D - 1> &coord) { this_ghost[coord] = 0; });
310  }
311  }
312  }
313  }
314 
315  // allocate recv buffers and post recvs
316  std::vector<std::vector<double>> recv_buffers(recv_buff_lengths.size());
317  for (size_t i = 0; i < recv_buff_lengths.size(); i++) {
318  recv_buffers[i].resize(recv_buff_lengths[i]);
319  }
320  std::vector<MPI_Request> recv_requests(recv_buff_lengths.size());
321  for (size_t i = 0; i < recv_requests.size(); i++) {
322  MPI_Irecv(recv_buffers[i].data(), recv_buffers[i].size(), MPI_DOUBLE, index_rank_map[i],
323  0, MPI_COMM_WORLD, &recv_requests[i]);
324  }
325 
326  // allocate send buffers
327  std::vector<std::vector<double>> out_buffers(send_buff_lengths.size());
328  for (size_t i = 0; i < send_buff_lengths.size(); i++) {
329  out_buffers[i].resize(send_buff_lengths[i]);
330  }
331  // perform fill operations for mpi buffers
332  std::vector<MPI_Request> send_requests(send_buff_lengths.size());
333  for (size_t i = 0; i < remote_calls.size(); i++) {
334  for (const RemoteCall &call : remote_calls[i]) {
335  auto pinfo = std::get<0>(call);
336  auto side = std::get<1>(call);
337  auto nbr_type = std::get<2>(call);
338  auto orthant = std::get<3>(call);
339  auto local_data = u->getLocalData(std::get<4>(call));
340  size_t buffer_offset = std::get<5>(call);
341  double *buffer_ptr = out_buffers[i].data() + buffer_offset;
342 
343  // create a LocalData object for the buffer
344  LocalData<D> buffer_data = getLocalDataForBuffer(buffer_ptr, side.opposite());
345 
346  // make the call
347  fillGhostCellsForNbrPatch(pinfo, local_data, buffer_data, side, nbr_type, orthant);
348  }
349  MPI_Isend(out_buffers[i].data(), out_buffers[i].size(), MPI_DOUBLE, index_rank_map[i],
350  0, MPI_COMM_WORLD, &send_requests[i]);
351  }
352  // perform local operations
353  for (auto pinfo : domain->getPatchInfoVector()) {
354  auto data = u->getLocalData(pinfo->local_index);
355  fillGhostCellsForLocalPatch(pinfo, data);
356  }
357  for (const LocalCall &call : local_calls) {
358  auto pinfo = std::get<0>(call);
359  auto side = std::get<1>(call);
360  auto nbr_type = std::get<2>(call);
361  auto orthant = std::get<3>(call);
362  auto local_data = u->getLocalData(std::get<4>(call));
363  auto nbr_data = u->getLocalData(std::get<5>(call));
364  fillGhostCellsForNbrPatch(pinfo, local_data, nbr_data, side, nbr_type, orthant);
365  }
366  // add in remote values as they come in
367  for (size_t i = 0; i < recv_requests.size(); i++) {
368  int finished_index;
369  MPI_Waitany(recv_requests.size(), recv_requests.data(), &finished_index,
370  MPI_STATUS_IGNORE);
371  for (auto t : incoming_ghosts[finished_index]) {
372  int local_index = std::get<0>(t);
373  Side<D> side = std::get<1>(t);
374  size_t buffer_offset = std::get<2>(t);
375 
376  const LocalData<D> local_data = u->getLocalData(local_index);
377  double * buffer_ptr = recv_buffers[finished_index].data() + buffer_offset;
378  LocalData<D> buffer_data = getLocalDataForBuffer(buffer_ptr, side);
379  for (int ig = 0; ig < domain->getNumGhostCells(); ig++) {
380  LocalData<D - 1> local_slice = local_data.getGhostSliceOnSide(side, ig + 1);
381  LocalData<D - 1> buffer_slice = buffer_data.getGhostSliceOnSide(side, ig + 1);
382  nested_loop<D - 1>(local_slice.getStart(), local_slice.getEnd(),
383  [&](const std::array<int, D - 1> &coord) {
384  local_slice[coord] += buffer_slice[coord];
385  });
386  }
387  }
388  }
389 
390  // wait for sends for finish
391  MPI_Waitall(send_requests.size(), send_requests.data(), MPI_STATUS_IGNORE);
392  }
393 };
