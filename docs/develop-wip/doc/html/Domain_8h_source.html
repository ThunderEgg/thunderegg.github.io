348  double integrate(std::shared_ptr<const Vector<D>> u) const
349  {
350  double sum = 0;
351 
352  for (auto &p : pinfo_id_map) {
353  PatchInfo<D> &d = *p.second;
354  for (int c = 0; c < u->getNumComponents(); c++) {
355  const LocalData<D> u_data = u->getLocalData(c, d.local_index);
356 
357  double patch_sum = 0;
358  nested_loop<D>(u_data.getStart(), u_data.getEnd(),
359  [&](std::array<int, D> coord) { patch_sum += u_data[coord]; });
360 
361  for (size_t i = 0; i < D; i++) {
362  patch_sum *= d.spacings[i];
363  }
364  sum += patch_sum;
365  }
366  }
367  double retval;
368  MPI_Allreduce(&sum, &retval, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
369  return retval;
370  }

418 template <int D> void Domain<D>::indexDomainsLocal(bool local_id_set)
419 {
420  std::vector<int> map_vec;
421  std::vector<int> in_off_proc_map_vec;
422  std::vector<int> out_off_proc_map_vec;
423  std::set<std::tuple<int, int>> in_offs;
424  std::set<std::tuple<int, int>> out_offs;
425  std::map<int, int> rev_map;
426  int curr_i = local_id_set ? pinfo_id_map.size() : 0;
427  if (local_id_set) {
428  if (!pinfo_id_map.empty()) {
429  std::set<int> todo;
430  map_vec.resize(pinfo_id_map.size());
431  for (auto &p : pinfo_id_map) {
432  todo.insert(p.first);
433  PatchInfo<D> &d = *p.second;
434  map_vec[d.local_index] = d.id;
435  }
436  std::set<int> enqueued;
437  while (!todo.empty()) {
438  std::deque<int> queue;
439  queue.push_back(*todo.begin());
440  enqueued.insert(*todo.begin());
441  while (!queue.empty()) {
442  int i = queue.front();
443  todo.erase(i);
444  queue.pop_front();
445  PatchInfo<D> &d = *pinfo_id_map[i];
446  rev_map[i] = d.local_index;
447  auto ranks = d.getNbrRanks();
448  auto ids = d.getNbrIds();
449  for (size_t idx = 0; idx < ranks.size(); idx++) {
450  int nbr_id = ids[idx];
451  int rank = ranks[idx];
452  if (pinfo_id_map.count(nbr_id)) {
453  if (!enqueued.count(nbr_id)) {
454  enqueued.insert(nbr_id);
455  queue.push_back(nbr_id);
456  }
457  } else {
458  in_offs.insert(std::make_tuple(rank, nbr_id));
459  out_offs.insert(std::make_tuple(rank, i));
460  }
461  }
462  }
463  }
464  }
465  } else {
466  std::set<int> offs;
467  if (!pinfo_id_map.empty()) {
468  std::set<int> todo;
469  for (auto &p : pinfo_id_map) {
470  todo.insert(p.first);
471  }
472  std::set<int> enqueued;
473  while (!todo.empty()) {
474  std::deque<int> queue;
475  queue.push_back(*todo.begin());
476  enqueued.insert(*todo.begin());
477  while (!queue.empty()) {
478  int i = queue.front();
479  todo.erase(i);
480  queue.pop_front();
481  map_vec.push_back(i);
482  PatchInfo<D> &d = *pinfo_id_map[i];
483  rev_map[i] = curr_i;
484  d.local_index = curr_i;
485  curr_i++;
486  auto ranks = d.getNbrRanks();
487  auto ids = d.getNbrIds();
488  for (size_t idx = 0; idx < ranks.size(); idx++) {
489  int nbr_id = ids[idx];
490  int rank = ranks[idx];
491  if (pinfo_id_map.count(nbr_id)) {
492  if (!enqueued.count(nbr_id)) {
493  enqueued.insert(nbr_id);
494  queue.push_back(nbr_id);
495  }
496  } else {
497  in_offs.insert(std::make_tuple(rank, nbr_id));
498  out_offs.insert(std::make_tuple(rank, i));
499  }
500  }
501  }
502  }
503  }
504  }
505  // map off proc
506  for (auto pair : in_offs) {
507  int rank = std::get<0>(pair);
508  int id = std::get<1>(pair);
509  if (in_off_proc_rank_vec.size() == 0 || rank != in_off_proc_rank_vec.back()) {
510  in_off_proc_rank_vec.push_back(rank);
511  in_off_proc_rank_size_vec.push_back(1);
512  } else {
513  in_off_proc_rank_size_vec.back()++;
514  }
515  in_off_proc_map_vec.push_back(id);
516  rev_map[id] = curr_i;
517  curr_i++;
518  }
519  for (auto pair : out_offs) {
520  int rank = std::get<0>(pair);
521  int id = std::get<1>(pair);
522  if (out_off_proc_rank_vec.size() == 0 || rank != out_off_proc_rank_vec.back()) {
523  out_off_proc_rank_vec.push_back(rank);
524  out_off_proc_rank_size_vec.push_back(1);
525  } else {
526  out_off_proc_rank_size_vec.back()++;
527  }
528  out_off_proc_map_vec.push_back(id);
529  }
530  // set local indexes
531  pinfo_vector.resize(pinfo_id_map.size());
532  for (auto &p : pinfo_id_map) {
533  p.second->setLocalNeighborIndexes(rev_map);
534  pinfo_vector[p.second->local_index] = p.second;
535  }
536  // domain_rev_map = rev_map;
537  patch_global_index_map_vec = map_vec;
538  patch_id_map_vec = map_vec;
539  this->in_off_proc_map_vec = in_off_proc_map_vec;
540  this->out_off_proc_map_vec = out_off_proc_map_vec;
541  patch_global_index_map_vec_off_proc = in_off_proc_map_vec;
542 }
543 template <int D> void Domain<D>::indexDomainsGlobal(bool global_id_set)
544 {
545  // global indices are going to be sequentially increasing with rank
546  int local_size = pinfo_id_map.size();
547  int start_i;
548  MPI_Scan(&local_size, &start_i, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
549  start_i -= local_size;
550  patch_global_index_map_vec.resize(local_size);
551  iota(patch_global_index_map_vec.begin(), patch_global_index_map_vec.end(), start_i);
552 
553  // prepare outgoing vector of data
554  std::vector<int> out_data = out_off_proc_map_vec;
555  for (size_t i = 0; i < out_data.size(); i++) {
556  out_data[i] = patch_global_index_map_vec[pinfo_id_map[out_data[i]]->local_index];
557  }
558  // send outgoing messages
559  using namespace std;
560  vector<MPI_Request> requests;
561 
562  // recv info
563  int curr_pos = 0;
564  for (size_t i = 0; i < in_off_proc_rank_vec.size(); i++) {
565  int source = in_off_proc_rank_vec[i];
566  int size = in_off_proc_rank_size_vec[i];
567  MPI_Request request;
568  MPI_Irecv(&patch_global_index_map_vec_off_proc[curr_pos], size, MPI_INT, source, 0,
569  MPI_COMM_WORLD, &request);
570  requests.push_back(request);
571  curr_pos += size;
572  }
573  // send info
574  curr_pos = 0;
575  requests.reserve(out_off_proc_rank_vec.size() + in_off_proc_rank_vec.size());
576  for (size_t i = 0; i < out_off_proc_rank_vec.size(); i++) {
577  int dest = out_off_proc_rank_vec[i];
578  int size = out_off_proc_rank_size_vec[i];
579  MPI_Request request;
580  MPI_Isend(&out_data[curr_pos], size, MPI_INT, dest, 0, MPI_COMM_WORLD, &request);
581  requests.push_back(request);
582  curr_pos += size;
583  }
584  // wait for all
585  MPI_Waitall(requests.size(), &requests[0], MPI_STATUSES_IGNORE);
586 
587  curr_pos = 0;
588  std::map<int, int> rev_map;
589  for (int i : patch_global_index_map_vec) {
590  rev_map[curr_pos] = i;
591  curr_pos++;
592  }
593  for (int i : patch_global_index_map_vec_off_proc) {
594  rev_map[curr_pos] = i;
595  curr_pos++;
596  }
597  for (auto &p : pinfo_id_map) {
598  p.second->setGlobalNeighborIndexes(rev_map);
599  }
600 }
601 template <int D> void Domain<D>::indexBCLocal()
602 {
603  int curr_i = 0;
604  for (auto pinfo : pinfo_vector) {
605  for (Side<D> s : Side<D>::getValues()) {
606  if (!pinfo->hasNbr(s)) {
607  pinfo->setBCLocalIndex(s, curr_i);
608  patch_id_bc_map_vec.push_back(pinfo->id);
609  curr_i++;
610  }
611  }
612  }
613  local_num_bc = curr_i;
