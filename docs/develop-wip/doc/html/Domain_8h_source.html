323  {
324  double sum = 0;

343  double integrate(std::shared_ptr<const Vector<D>> u) const
344  {
345  double sum = 0;
346 
347  for (auto &p : pinfo_id_map) {
348  PatchInfo<D> & d = *p.second;
349  const LocalData<D> u_data = u->getLocalData(d.local_index);
350 
351  double patch_sum = 0;
352  nested_loop<D>(u_data.getStart(), u_data.getEnd(),
353  [&](std::array<int, D> coord) { patch_sum += u_data[coord]; });
354 
355  for (size_t i = 0; i < D; i++) {
356  patch_sum *= d.spacings[i];
357  }
358  sum += patch_sum;
359  }
360  double retval;
361  MPI_Allreduce(&sum, &retval, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
362  return retval;
363  }

366 template <size_t D> void Domain<D>::indexDomainsLocal(bool local_id_set)
367 {
368  std::vector<int> map_vec;
369  std::vector<int> in_off_proc_map_vec;
370  std::vector<int> out_off_proc_map_vec;
371  std::set<std::tuple<int, int>> in_offs;
372  std::set<std::tuple<int, int>> out_offs;
373  std::map<int, int> rev_map;
374  int curr_i = local_id_set ? pinfo_id_map.size() : 0;
375  if (local_id_set) {
376  if (!pinfo_id_map.empty()) {
377  std::set<int> todo;
378  map_vec.resize(pinfo_id_map.size());
379  for (auto &p : pinfo_id_map) {
380  todo.insert(p.first);
381  PatchInfo<D> &d = *p.second;
382  map_vec[d.local_index] = d.id;
383  }
384  std::set<int> enqueued;
385  while (!todo.empty()) {
386  std::deque<int> queue;
387  queue.push_back(*todo.begin());
388  enqueued.insert(*todo.begin());
389  while (!queue.empty()) {
390  int i = queue.front();
391  todo.erase(i);
392  queue.pop_front();
393  PatchInfo<D> &d = *pinfo_id_map[i];
394  rev_map[i] = d.local_index;
395  auto ranks = d.getNbrRanks();
396  auto ids = d.getNbrIds();
397  for (size_t idx = 0; idx < ranks.size(); idx++) {
398  int nbr_id = ids[idx];
399  int rank = ranks[idx];
400  if (pinfo_id_map.count(nbr_id)) {
401  if (!enqueued.count(nbr_id)) {
402  enqueued.insert(nbr_id);
403  queue.push_back(nbr_id);
404  }
405  } else {
406  in_offs.insert(std::make_tuple(rank, nbr_id));
407  out_offs.insert(std::make_tuple(rank, i));
408  }
409  }
410  }
411  }
412  }
413  } else {
414  std::set<int> offs;
415  if (!pinfo_id_map.empty()) {
416  std::set<int> todo;
417  for (auto &p : pinfo_id_map) {
418  todo.insert(p.first);
419  }
420  std::set<int> enqueued;
421  while (!todo.empty()) {
422  std::deque<int> queue;
423  queue.push_back(*todo.begin());
424  enqueued.insert(*todo.begin());
425  while (!queue.empty()) {
426  int i = queue.front();
427  todo.erase(i);
428  queue.pop_front();
429  map_vec.push_back(i);
430  PatchInfo<D> &d = *pinfo_id_map[i];
431  rev_map[i] = curr_i;
432  d.local_index = curr_i;
433  curr_i++;
434  auto ranks = d.getNbrRanks();
435  auto ids = d.getNbrIds();
436  for (size_t idx = 0; idx < ranks.size(); idx++) {
437  int nbr_id = ids[idx];
438  int rank = ranks[idx];
439  if (pinfo_id_map.count(nbr_id)) {
440  if (!enqueued.count(nbr_id)) {
441  enqueued.insert(nbr_id);
442  queue.push_back(nbr_id);
443  }
444  } else {
445  in_offs.insert(std::make_tuple(rank, nbr_id));
446  out_offs.insert(std::make_tuple(rank, i));
447  }
448  }
449  }
450  }
451  }
452  }
453  // map off proc
454  for (auto pair : in_offs) {
455  int rank = std::get<0>(pair);
456  int id = std::get<1>(pair);
457  if (in_off_proc_rank_vec.size() == 0 || rank != in_off_proc_rank_vec.back()) {
458  in_off_proc_rank_vec.push_back(rank);
459  in_off_proc_rank_size_vec.push_back(1);
460  } else {
461  in_off_proc_rank_size_vec.back()++;
462  }
463  in_off_proc_map_vec.push_back(id);
464  rev_map[id] = curr_i;
465  curr_i++;
466  }
467  for (auto pair : out_offs) {
468  int rank = std::get<0>(pair);
469  int id = std::get<1>(pair);
470  if (out_off_proc_rank_vec.size() == 0 || rank != out_off_proc_rank_vec.back()) {
471  out_off_proc_rank_vec.push_back(rank);
472  out_off_proc_rank_size_vec.push_back(1);
473  } else {
474  out_off_proc_rank_size_vec.back()++;
475  }
476  out_off_proc_map_vec.push_back(id);
477  }
478  // set local indexes
479  pinfo_vector.resize(pinfo_id_map.size());
480  for (auto &p : pinfo_id_map) {
481  p.second->setLocalNeighborIndexes(rev_map);
482  pinfo_vector[p.second->local_index] = p.second;
483  }
484  // domain_rev_map = rev_map;
485  patch_global_index_map_vec = map_vec;
486  patch_id_map_vec = map_vec;
487  this->in_off_proc_map_vec = in_off_proc_map_vec;
488  this->out_off_proc_map_vec = out_off_proc_map_vec;
489  patch_global_index_map_vec_off_proc = in_off_proc_map_vec;
490 }
491 template <size_t D> void Domain<D>::indexDomainsGlobal(bool global_id_set)
492 {
493  // global indices are going to be sequentially increasing with rank
494  int local_size = pinfo_id_map.size();
495  int start_i;
496  MPI_Scan(&local_size, &start_i, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
497  start_i -= local_size;
498  patch_global_index_map_vec.resize(local_size);
499  iota(patch_global_index_map_vec.begin(), patch_global_index_map_vec.end(), start_i);
500 
501  // prepare outgoing vector of data
502  std::vector<int> out_data = out_off_proc_map_vec;
503  for (size_t i = 0; i < out_data.size(); i++) {
504  out_data[i] = patch_global_index_map_vec[pinfo_id_map[out_data[i]]->local_index];
505  }
506  // send outgoing messages
507  using namespace std;
508  vector<MPI_Request> requests;
509 
510  // recv info
511  int curr_pos = 0;
512  for (size_t i = 0; i < in_off_proc_rank_vec.size(); i++) {
513  int source = in_off_proc_rank_vec[i];
514  int size = in_off_proc_rank_size_vec[i];
515  MPI_Request request;
516  MPI_Irecv(&patch_global_index_map_vec_off_proc[curr_pos], size, MPI_INT, source, 0,
517  MPI_COMM_WORLD, &request);
518  requests.push_back(request);
519  curr_pos += size;
520  }
521  // send info
522  curr_pos = 0;
523  requests.reserve(out_off_proc_rank_vec.size() + in_off_proc_rank_vec.size());
524  for (size_t i = 0; i < out_off_proc_rank_vec.size(); i++) {
525  int dest = out_off_proc_rank_vec[i];
526  int size = out_off_proc_rank_size_vec[i];
527  MPI_Request request;
528  MPI_Isend(&out_data[curr_pos], size, MPI_INT, dest, 0, MPI_COMM_WORLD, &request);
529  requests.push_back(request);
530  curr_pos += size;
531  }
532  // wait for all
533  MPI_Waitall(requests.size(), &requests[0], MPI_STATUSES_IGNORE);
534 
535  curr_pos = 0;
536  std::map<int, int> rev_map;
537  for (int i : patch_global_index_map_vec) {
538  rev_map[curr_pos] = i;
539  curr_pos++;
540  }
541  for (int i : patch_global_index_map_vec_off_proc) {
542  rev_map[curr_pos] = i;
543  curr_pos++;
544  }
545  for (auto &p : pinfo_id_map) {
546  p.second->setGlobalNeighborIndexes(rev_map);
547  }
548 }
549 template <size_t D> void Domain<D>::indexBCLocal()
550 {
551  int curr_i = 0;
552  for (auto pinfo : pinfo_vector) {
553  for (Side<D> s : Side<D>::getValues()) {
554  if (!pinfo->hasNbr(s)) {
555  pinfo->setBCLocalIndex(s, curr_i);
556  patch_id_bc_map_vec.push_back(pinfo->id);
557  curr_i++;
558  }
559  }
560  }
561  local_num_bc = curr_i;
562 }

564 {

569  for (Side<D> s : Side<D>::getValues()) {

571  pinfo->setBCGlobalIndex(s, curr_i);
572  curr_i++;
573  }
574  }
