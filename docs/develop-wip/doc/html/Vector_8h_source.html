241 inline LocalData<D - 1> LocalData<D>::getSliceOnSidePriv(Side<D> s, int offset) const
242 {
243  size_t axis = s.getAxisIndex();
244  std::array<int, D - 1> new_strides;
245  for (size_t i = 0; i < axis; i++) {
246  new_strides[i] = strides[i];
247  }
248  for (size_t i = axis; i < D - 1; i++) {
249  new_strides[i] = strides[i + 1];
250  }
251  std::array<int, D - 1> new_lengths;
252  for (size_t i = 0; i < axis; i++) {
253  new_lengths[i] = lengths[i];
254  }
255  for (size_t i = axis; i < D - 1; i++) {
256  new_lengths[i] = lengths[i + 1];
257  }
258  if (s.isLowerOnAxis()) {
259  double *new_data = data + offset * strides[axis];
260  return LocalData<D - 1>(new_data, new_strides, new_lengths, 0, ldm);
261  } else {
262  double *new_data = data + (lengths[axis] - 1 - offset) * strides[axis];
263  return LocalData<D - 1>(new_data, new_strides, new_lengths, 0, ldm);
264  }
265 }
