332  virtual double twoNorm() const
333  {
334  double sum = 0;
335  for (int i = 0; i < num_local_patches; i++) {
336  const std::vector<LocalData<D>> lds = getLocalDatas(i);
337  for (const auto &ld : lds) {
338  nested_loop<D>(ld.getStart(), ld.getEnd(),
339  [&](std::array<int, D> coord) { sum += ld[coord] * ld[coord]; });
340  }

344  return sqrt(global_sum);
