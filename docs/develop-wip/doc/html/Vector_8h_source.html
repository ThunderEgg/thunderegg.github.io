240 inline LocalData<D - 1> LocalData<D>::getSliceOnSidePriv(Side<D> s, int offset) const
241 {
242  size_t axis = s.getAxisIndex();
243  std::array<int, D - 1> new_strides;
244  for (size_t i = 0; i < axis; i++) {
245  new_strides[i] = strides[i];
246  }
247  for (size_t i = axis; i < D - 1; i++) {
248  new_strides[i] = strides[i + 1];
249  }
250  std::array<int, D - 1> new_lengths;
251  for (size_t i = 0; i < axis; i++) {
252  new_lengths[i] = lengths[i];
253  }
254  for (size_t i = axis; i < D - 1; i++) {
255  new_lengths[i] = lengths[i + 1];
256  }
257  if (s.isLowerOnAxis()) {
258  double *new_data = data + offset * strides[axis];
259  return LocalData<D - 1>(new_data, new_strides, new_lengths, 0, ldm);
260  } else {
261  double *new_data = data + (lengths[axis] - 1 - offset) * strides[axis];
262  return LocalData<D - 1>(new_data, new_strides, new_lengths, 0, ldm);
263  }
