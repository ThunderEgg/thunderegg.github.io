82  static int GetNumLocalPatches(Vec vec, const std::array<int, D> &lengths, int num_ghost_cells,
83  int num_components)

103  static int GetNumLocalCells(Vec vec, const std::array<int, D> &lengths, int num_ghost_cells,
104  int num_components)

124  VecWrapper(Vec vec, const std::array<int, D> &lengths, int num_components, int num_ghost_cells,
125  bool own)
126  : Vector<D>(MPI_COMM_WORLD, num_components,
127  GetNumLocalPatches(vec, lengths, num_ghost_cells, num_components),
128  GetNumLocalCells(vec, lengths, num_ghost_cells, num_components)),
129  vec(vec), num_ghost_cells(num_ghost_cells), own(own), lengths(lengths)
130  {
131  strides[0] = 1;
132  int my_first_offset = num_ghost_cells;
133  for (size_t i = 1; i < D; i++) {
134  strides[i] = (this->lengths[i - 1] + 2 * num_ghost_cells) * strides[i - 1];
135  my_first_offset += strides[i] * num_ghost_cells;
136  }
137  first_offset = my_first_offset;
138  component_stride = strides[D - 1] * (lengths[D - 1] + 2 * num_ghost_cells);
139  patch_stride = component_stride * num_components;
140  }
