28 template <size_t D> class ValVector : public Vector<D>
29 
30 {
31  private:
32  int patch_stride;

34  std::array<int, D> strides;
35  int num_ghost_cells;
36  int first_offset;
37 
38  public:
39  std::valarray<double> vec;
40  std::valarray<double> ghost_data;
41  ValVector(MPI_Comm comm_in, const std::array<int, D> &lengths, int num_ghost_cells,
42  int num_patches)
43  : Vector<D>(comm_in)
44  {
45  this->num_local_patches = num_patches;
46  this->num_ghost_cells = num_ghost_cells;
47  int size = 1;
48  this->lengths = lengths;
49  first_offset = 0;
50  for (size_t i = 0; i < D; i++) {
51  strides[i] = size;
52  size *= (this->lengths[i] + 2 * num_ghost_cells);
53  first_offset += strides[i] * num_ghost_cells;
54  }
55  patch_stride = size;
56  size *= num_patches;
57  vec.resize(size);
58 
59  int num_cells_in_patch = 1;
60  for (size_t i = 0; i < D; i++) {
61  num_cells_in_patch *= this->lengths[i];
62  }
63  this->num_local_cells = this->num_local_patches * num_cells_in_patch;
64  }
65  ~ValVector() = default;
66  static std::shared_ptr<ValVector<D>> GetNewVector(std::shared_ptr<const Domain<D>> domain)
67  {
68  return std::shared_ptr<ValVector<D>>(new ValVector<D>(
69  MPI_COMM_WORLD, domain->getNs(), domain->getNumGhostCells(), domain->getNumLocalPatches()));
70  }
71  LocalData<D> getLocalData(int local_patch_id)
72  {
73  double *data = &vec[patch_stride * local_patch_id + first_offset];
74  return LocalData<D>(data, strides, lengths, num_ghost_cells, nullptr);
75  }
76  const LocalData<D> getLocalData(int local_patch_id) const
77  {
78  double *data = const_cast<double *>(&vec[patch_stride * local_patch_id + first_offset]);
79  return LocalData<D>(data, strides, lengths, num_ghost_cells, nullptr);
80  }
81  void setNumGhostPatches(int num_ghost_patches)
82  {
83  ghost_data.resize(patch_stride * num_ghost_patches);
84  }
85 };
