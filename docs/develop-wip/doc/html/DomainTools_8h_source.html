42  static void getRealCoord(std::shared_ptr<const PatchInfo<D>> pinfo,
43  const std::array<int, D> &coord, std::array<double, D> &real_coord)
44  {
45  loop<0, D - 1>([&](int dir) {
46  if (coord[dir] == -1) {
47  real_coord[dir] = pinfo->starts[dir];
48  } else if (coord[dir] == pinfo->ns[dir]) {
49  real_coord[dir] = pinfo->starts[dir] + pinfo->spacings[dir] * pinfo->ns[dir];
50  } else {
51  real_coord[dir] = pinfo->starts[dir] + pinfo->spacings[dir] / 2.0
52  + pinfo->spacings[dir] * coord[dir];
53  }
54  });
55  }
59  static void getRealCoordGhost(std::shared_ptr<const PatchInfo<D>> pinfo,

72  static void getRealCoordBound(std::shared_ptr<const PatchInfo<D>> pinfo,
73  const std::array<int, D - 1> &coord, Side<D> s,
74  std::array<double, D> &real_coord)
75  {
76  for (size_t dir = 0; dir < s.getAxisIndex(); dir++) {
77  if (coord[dir] == -1) {
78  real_coord[dir] = pinfo->starts[dir];
79  } else if (coord[dir] == pinfo->ns[dir]) {
80  real_coord[dir] = pinfo->starts[dir] + pinfo->spacings[dir] * pinfo->ns[dir];
81  } else {
82  real_coord[dir] = pinfo->starts[dir] + pinfo->spacings[dir] / 2.0
83  + pinfo->spacings[dir] * coord[dir];
84  }
85  }
86  if (s.isLowerOnAxis()) {
87  real_coord[s.getAxisIndex()] = pinfo->starts[s.getAxisIndex()];
88  } else {
89  real_coord[s.getAxisIndex()]
90  = pinfo->starts[s.getAxisIndex()]
91  + pinfo->spacings[s.getAxisIndex()] * pinfo->ns[s.getAxisIndex()];
92  }
93  for (size_t dir = s.getAxisIndex() + 1; dir < D; dir++) {
94  if (coord[dir - 1] == -1) {
95  real_coord[dir] = pinfo->starts[dir];
96  } else if (coord[dir - 1] == pinfo->ns[dir]) {
97  real_coord[dir] = pinfo->starts[dir] + pinfo->spacings[dir] * pinfo->ns[dir];
98  } else {
99  real_coord[dir] = pinfo->starts[dir] + pinfo->spacings[dir] / 2.0
100  + pinfo->spacings[dir] * coord[dir - 1];
101  }
102  }
103  }
107  static void setValues(std::shared_ptr<Domain<D>> domain, std::shared_ptr<Vector<D>> vec,
108  std::function<double(const std::array<double, D> &)> func)
109  {
110  std::array<double, D> real_coord;
111  for (int i = 0; i < vec->getNumLocalPatches(); i++) {
112  LocalData<D> ld = vec->getLocalData(i);
113  auto pinfo = domain->getPatchInfoVector()[i];
114  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
115  getRealCoord(pinfo, coord, real_coord);
116  ld[coord] = func(real_coord);
117  });
118  }
119  }
