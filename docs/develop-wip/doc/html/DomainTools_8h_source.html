46 template <int D>
47 void GetRealCoord(std::shared_ptr<const PatchInfo<D>> pinfo, const std::array<int, D> &coord,
48  std::array<double, D> &real_coord)
49 {
50  loop<0, D - 1>([&](int dir) {
51  if (coord[dir] == -1) {
52  real_coord[dir] = pinfo->starts[dir];
53  } else if (coord[dir] == pinfo->ns[dir]) {
54  real_coord[dir] = pinfo->starts[dir] + pinfo->spacings[dir] * pinfo->ns[dir];
55  } else {
56  real_coord[dir]
57  = pinfo->starts[dir] + pinfo->spacings[dir] / 2.0 + pinfo->spacings[dir] * coord[dir];
58  }
59  });
60 }

88 template <int D>
89 void GetRealCoordBound(std::shared_ptr<const PatchInfo<D>> pinfo,
90  const std::array<int, D - 1> &coord, Side<D> s,
91  std::array<double, D> &real_coord)
92 {
93  for (size_t dir = 0; dir < s.getAxisIndex(); dir++) {
94  if (coord[dir] == -1) {
95  real_coord[dir] = pinfo->starts[dir];
96  } else if (coord[dir] == pinfo->ns[dir]) {
97  real_coord[dir] = pinfo->starts[dir] + pinfo->spacings[dir] * pinfo->ns[dir];
98  } else {
99  real_coord[dir]
100  = pinfo->starts[dir] + pinfo->spacings[dir] / 2.0 + pinfo->spacings[dir] * coord[dir];
101  }
102  }
103  if (s.isLowerOnAxis()) {
104  real_coord[s.getAxisIndex()] = pinfo->starts[s.getAxisIndex()];
105  } else {
106  real_coord[s.getAxisIndex()]
107  = pinfo->starts[s.getAxisIndex()]
108  + pinfo->spacings[s.getAxisIndex()] * pinfo->ns[s.getAxisIndex()];
109  }
110  for (size_t dir = s.getAxisIndex() + 1; dir < D; dir++) {
111  if (coord[dir - 1] == -1) {
112  real_coord[dir] = pinfo->starts[dir];
113  } else if (coord[dir - 1] == pinfo->ns[dir]) {
114  real_coord[dir] = pinfo->starts[dir] + pinfo->spacings[dir] * pinfo->ns[dir];
115  } else {
116  real_coord[dir] = pinfo->starts[dir] + pinfo->spacings[dir] / 2.0
117  + pinfo->spacings[dir] * coord[dir - 1];
118  }
119  }
120 }
131 template <int D, typename T>
132 void SetValues(std::shared_ptr<Domain<D>> domain, std::shared_ptr<Vector<D>> vec,
133  int component_index, T func)
134 {
135  if (component_index >= vec->getNumComponents()) {
136  throw RuntimeError("Invalid component to set");
137  }
138  std::array<double, D> real_coord;
139  for (int i = 0; i < vec->getNumLocalPatches(); i++) {
140  LocalData<D> ld = vec->getLocalData(component_index, i);
141  auto pinfo = domain->getPatchInfoVector()[i];
142  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
143  GetRealCoord<D>(pinfo, coord, real_coord);
144  ld[coord] = func(real_coord);
145  });
146  }
147 }

290 template <int D, typename T>
291 void SetBCValues(std::shared_ptr<Domain<D>> domain, std::shared_ptr<Vector<D - 1>> vec, T func,
292  int component_index = 0)
293 {
294  if (component_index >= vec->getNumComponents()) {
295  throw RuntimeError("More functions given than available components");
296  }
297  std::array<double, D> real_coord;
298  for (int i = 0; i < vec->getNumLocalPatches(); i++) {
299  auto pinfo = domain->getPatchInfoMap()[domain->patch_id_bc_map_vec[i]];
300  for (Side<D> s : Side<D>::getValues()) {
301  if (!pinfo->hasNbr(s)) {
302  LocalData<D - 1> ld = vec->getLocalData(component_index, pinfo->getBCLocalIndex(s));
303  nested_loop<D - 1>(ld.getStart(), ld.getEnd(),
304  [&](const std::array<int, D - 1> &coord) {
305  GetRealCoordBound<D>(pinfo, coord, s, real_coord);
306  ld[coord] = func(real_coord);
307  });
308  }
309  }
