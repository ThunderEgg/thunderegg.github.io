91  std::array<fftw_r2r_kind, D>
92  getTransformsForPatch(std::shared_ptr<const ThunderEgg::PatchInfo<D>> pinfo)
93  {
94  // get transform types for each axis
95  std::array<fftw_r2r_kind, D> transforms;
96  for (size_t axis = 0; axis < D; axis++) {
97  if (pinfo->isNeumann(Side<D>::LowerSideOnAxis(axis))
98  && pinfo->isNeumann(Side<D>::HigherSideOnAxis(axis))) {
99  transforms[D - 1 - axis] = FFTW_REDFT10;
100  } else if (pinfo->isNeumann(Side<D>::LowerSideOnAxis(axis))) {
101  transforms[D - 1 - axis] = FFTW_REDFT11;
102  } else if (pinfo->isNeumann(Side<D>::HigherSideOnAxis(axis))) {
103  transforms[D - 1 - axis] = FFTW_RODFT11;
104  } else {
105  transforms[D - 1 - axis] = FFTW_RODFT10;
106  }
107  }
108  return transforms;
109  }

142  std::valarray<double> getEigenValues(std::shared_ptr<const PatchInfo<D>> pinfo)
143  {
144  std::valarray<double> retval(this->domain->getNumCellsInPatch());
145 
146  std::valarray<size_t> all_strides(D);
147  size_t curr_stride = 1;
148  for (size_t i = 0; i < D; i++) {
149  all_strides[i] = curr_stride;
150  curr_stride *= pinfo->ns[i];
151  }
152 
153  for (size_t axis = 0; axis < D; axis++) {
154  int n = pinfo->ns[axis];
155  double h = pinfo->spacings[axis];
156 
157  std::valarray<size_t> sizes(D - 1);
158  std::valarray<size_t> strides(D - 1);
159  size_t ones_size = 1;
160  for (size_t i = 0; i < axis; i++) {
161  strides[i] = all_strides[i];
162  sizes[i] = pinfo->ns[i];
163  ones_size *= pinfo->ns[i];
164  }
165  int input_stride = (int) all_strides[axis];
166  for (size_t i = axis + 1; i < D; i++) {
167  strides[i - 1] = all_strides[i];
168  sizes[i - 1] = pinfo->ns[i];
169  ones_size *= pinfo->ns[i];
170  }
171 
172  std::valarray<double> ones(ones_size);
173  ones = 1;
174 
175  if (pinfo->isNeumann(Side<D>::LowerSideOnAxis(axis))
176  && pinfo->isNeumann(Side<D>::HigherSideOnAxis(axis))) {
177  for (int xi = 0; xi < n; xi++) {
178  retval[std::gslice(xi * input_stride, sizes, strides)]
179  -= 4 / (h * h) * pow(sin(xi * M_PI / (2 * n)), 2) * ones;
180  }
181  } else if (pinfo->isNeumann(Side<D>::LowerSideOnAxis(axis))
182  || pinfo->isNeumann(Side<D>::HigherSideOnAxis(axis))) {
183  for (int xi = 0; xi < n; xi++) {
184  retval[std::gslice(xi * input_stride, sizes, strides)]
185  -= 4 / (h * h) * pow(sin((xi + 0.5) * M_PI / (2 * n)), 2) * ones;
186  }
187  } else {
188  for (int xi = 0; xi < n; xi++) {
189  retval[std::gslice(xi * input_stride, sizes, strides)]
190  -= 4 / (h * h) * pow(sin((xi + 1) * M_PI / (2 * n)), 2) * ones;
191  }
192  }
193  }
194  return retval;
195  }

214  void solveSinglePatch(std::shared_ptr<const PatchInfo<D>> pinfo,
215  const std::vector<LocalData<D>> & fs,
216  std::vector<LocalData<D>> & us) const override
217  {
218  LocalData<D> f_copy_ld = f_copy->getLocalData(0, 0);
219 
220  nested_loop<D>(f_copy_ld.getStart(), f_copy_ld.getEnd(),
221  [&](std::array<int, D> coord) { f_copy_ld[coord] = fs[0][coord]; });
222 
223  std::vector<LocalData<D>> f_copy_lds = {f_copy_ld};
224  op->addGhostToRHS(pinfo, us, f_copy_lds);
225 
226  fftw_execute(plan1.at(pinfo));
227 
228  tmp->getValArray() /= eigen_vals.at(pinfo);
229 
230  if (pinfo->neumann.all()) {
231  tmp->getValArray()[0] = 0;
232  }
233 
234  fftw_execute(plan2.at(pinfo));
235 
236  LocalData<D> sol_ld = sol->getLocalData(0, 0);
237 
238  double scale = 1;
239  for (size_t axis = 0; axis < D; axis++) {
240  scale *= 2.0 * this->domain->getNs()[axis];
241  }
242  nested_loop<D>(us[0].getStart(), us[0].getEnd(),
243  [&](std::array<int, D> coord) { us[0][coord] = sol_ld[coord] / scale; });
244  }
252  void addPatch(std::shared_ptr<const PatchInfo<D>> pinfo)
253  {
254  if (plan1.count(pinfo) == 0) {
255  // revers ns because FFTW is row major
256  std::array<int, D> ns_reversed;
257  for (size_t i = 0; i < D; i++) {
258  ns_reversed[D - 1 - i] = pinfo->ns[i];
259  }
260  std::array<fftw_r2r_kind, D> transforms = getTransformsForPatch(pinfo);
261  std::array<fftw_r2r_kind, D> transforms_inv = getInverseTransformsForPatch(pinfo);
262 
263  plan1[pinfo] = fftw_plan_r2r(D, ns_reversed.data(), &f_copy->getValArray()[0],
264  &tmp->getValArray()[0], transforms.data(),
265  FFTW_MEASURE | FFTW_DESTROY_INPUT);
266  plan2[pinfo]
267  = fftw_plan_r2r(D, ns_reversed.data(), &tmp->getValArray()[0], &sol->getValArray()[0],
268  transforms_inv.data(), FFTW_MEASURE | FFTW_DESTROY_INPUT);
269 
270  eigen_vals[pinfo] = getEigenValues(pinfo);
271  }
272  }
