805 template <size_t D> inline void PatchInfo<D>::setNeumann(IsNeumannFunc<D> inf)
806 {
807  for (Side<D> s : Side<D>::getValues()) {
808  if (!hasNbr(s)) {
809  std::array<double, D> bound_start = starts;
810  if (!s.isLowerOnAxis()) {
811  bound_start[s.getAxisIndex()] += spacings[s.getAxisIndex()] * ns[s.getAxisIndex()];
812  }
813  std::array<double, D> bound_end = bound_start;
814  for (size_t dir = 0; dir < D; dir++) {
815  if (dir != s.getAxisIndex()) {
816  bound_end[dir] += spacings[dir] * ns[dir];
817  }
818  }
819  neumann[s.getIndex()] = inf(s, bound_end, bound_start);
820  }
821  }

845 {
846  BufferWriter writer(buffer);
847  writer << id;

849  writer << refine_level;

853  writer << child_ranks;
854  writer << rank;
855  writer << orth_on_parent;
856  writer << neumann;
857  writer << starts;
858  writer << spacings;
859  std::bitset<Side<D>::num_sides> has_nbr;
860  for (size_t i = 0; i < Side<D>::num_sides; i++) {
861  has_nbr[i] = nbr_info[i] != nullptr;
862  }
863  writer << has_nbr;
864  for (Side<D> s : Side<D>::getValues()) {
865  if (hasNbr(s)) {
866  NbrType type = getNbrType(s);
867  writer << type;
868  switch (type) {
869  case NbrType::Normal: {
870  NormalNbrInfo<D> tmp = getNormalNbrInfo(s);
871  writer << tmp;
872  } break;
873  case NbrType::Fine: {
874  FineNbrInfo<D> tmp = getFineNbrInfo(s);
875  writer << tmp;
876  } break;
877  case NbrType::Coarse: {
878  CoarseNbrInfo<D> tmp = getCoarseNbrInfo(s);
879  writer << tmp;
880  } break;
881  }
882  }
883  }
884  return writer.getPos();

886 template <size_t D> inline int PatchInfo<D>::deserialize(char *buffer)
887 {
888  BufferReader reader(buffer);
889  reader >> id;
890  reader >> ns;
891  reader >> refine_level;
892  reader >> parent_id;
893  reader >> parent_rank;
894  reader >> child_ids;
895  reader >> child_ranks;
896  reader >> rank;
897  reader >> orth_on_parent;
898  reader >> neumann;
899  reader >> starts;
900  reader >> spacings;
901  std::bitset<Side<D>::num_sides> has_nbr;
902  reader >> has_nbr;
903  for (size_t i = 0; i < Side<D>::num_sides; i++) {
904  if (has_nbr[i]) {
905  NbrType type;
906  reader >> type;
907  std::shared_ptr<NbrInfo<D>> info;
908  switch (type) {
909  case NbrType::Normal:
910  info.reset(new NormalNbrInfo<D>());
911  reader >> *std::static_pointer_cast<NormalNbrInfo<D>>(info);
912  break;
913  case NbrType::Fine:
914  info.reset(new FineNbrInfo<D>());
915  reader >> *std::static_pointer_cast<FineNbrInfo<D>>(info);
916  break;
917  case NbrType::Coarse:
918  info.reset(new CoarseNbrInfo<D>());
919  reader >> *std::static_pointer_cast<CoarseNbrInfo<D>>(info);
920  break;
921  }
922  nbr_info[i] = info;
923  }
924  }
925  return reader.getPos();
926 }
