692 template <size_t D> inline void PatchInfo<D>::setNeumann(IsNeumannFunc<D> inf)
693 {
694  for (Side<D> s : Side<D>::getValues()) {
695  if (!hasNbr(s)) {
696  std::array<double, D> bound_start = starts;
697  if (!s.isLowerOnAxis()) {
698  bound_start[s.getAxisIndex()] += spacings[s.getAxisIndex()] * ns[s.getAxisIndex()];
699  }
700  std::array<double, D> bound_end = bound_start;
701  for (size_t dir = 0; dir < D; dir++) {
702  if (dir != s.getAxisIndex()) {
703  bound_end[dir] += spacings[dir] * ns[dir];
704  }
705  }
706  neumann[s.getIndex()] = inf(s, bound_end, bound_start);
707  }
708  }

731 template <size_t D> inline int PatchInfo<D>::serialize(char *buffer) const
732 {
733  BufferWriter writer(buffer);
734  writer << id;
735  writer << ns;
736  writer << refine_level;
737  writer << parent_id;
738  writer << parent_rank;
739  writer << child_ids;
740  writer << child_ranks;
741  writer << rank;
742  writer << orth_on_parent;
743  writer << neumann;
744  writer << starts;
745  writer << spacings;
746  std::bitset<Side<D>::num_sides> has_nbr;
747  for (size_t i = 0; i < Side<D>::num_sides; i++) {
748  has_nbr[i] = nbr_info[i] != nullptr;
749  }
750  writer << has_nbr;
751  for (Side<D> s : Side<D>::getValues()) {
752  if (hasNbr(s)) {
753  NbrType type = getNbrType(s);
754  writer << type;
755  switch (type) {
756  case NbrType::Normal: {
757  NormalNbrInfo<D> tmp = getNormalNbrInfo(s);
758  writer << tmp;
759  } break;
760  case NbrType::Fine: {
761  FineNbrInfo<D> tmp = getFineNbrInfo(s);
762  writer << tmp;
763  } break;
764  case NbrType::Coarse: {
765  CoarseNbrInfo<D> tmp = getCoarseNbrInfo(s);
766  writer << tmp;
767  } break;
768  }
769  }
770  }
771  return writer.getPos();
772 }
773 template <size_t D> inline int PatchInfo<D>::deserialize(char *buffer)
774 {
775  BufferReader reader(buffer);
776  reader >> id;
777  reader >> ns;
778  reader >> refine_level;
779  reader >> parent_id;
780  reader >> parent_rank;
781  reader >> child_ids;
782  reader >> child_ranks;
783  reader >> rank;
784  reader >> orth_on_parent;
785  reader >> neumann;
786  reader >> starts;
787  reader >> spacings;
788  std::bitset<Side<D>::num_sides> has_nbr;
789  reader >> has_nbr;
790  for (size_t i = 0; i < Side<D>::num_sides; i++) {
791  if (has_nbr[i]) {
792  NbrType type;
793  reader >> type;
794  std::shared_ptr<NbrInfo<D>> info;
795  switch (type) {
796  case NbrType::Normal:
797  info.reset(new NormalNbrInfo<D>());
798  reader >> *std::static_pointer_cast<NormalNbrInfo<D>>(info);
799  break;
800  case NbrType::Fine:
801  info.reset(new FineNbrInfo<D>());
802  reader >> *std::static_pointer_cast<FineNbrInfo<D>>(info);
803  break;
804  case NbrType::Coarse:
805  info.reset(new CoarseNbrInfo<D>());
806  reader >> *std::static_pointer_cast<CoarseNbrInfo<D>>(info);
807  break;
808  }
809  nbr_info[i] = info;
810  }
811  }
812  return reader.getPos();
813 }
