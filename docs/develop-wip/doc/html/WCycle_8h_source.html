47  void visit(const Level<D> &level, std::list<std::shared_ptr<Vector<D>>> &u_vectors,
48  std::list<std::shared_ptr<const Vector<D>>> &f_vectors) const
49  {
50  if (level.coarsest()) {
51  for (int i = 0; i < num_coarse_sweeps; i++) {
52  this->smooth(level, u_vectors, f_vectors);
53  }
54  } else {
55  for (int i = 0; i < num_pre_sweeps; i++) {
56  this->smooth(level, u_vectors, f_vectors);
57  }
58  this->prepCoarser(level, u_vectors, f_vectors);
59  this->visit(*level.getCoarser(), u_vectors, f_vectors);
60  for (int i = 0; i < num_mid_sweeps; i++) {
61  this->smooth(level, u_vectors, f_vectors);
62  }
63  this->prepCoarser(level, u_vectors, f_vectors);
64  this->visit(*level.getCoarser(), u_vectors, f_vectors);
65  for (int i = 0; i < num_post_sweeps; i++) {
66  this->smooth(level, u_vectors, f_vectors);
67  }
68  }
69  if (!level.finest()) { this->prepFiner(level, u_vectors, f_vectors); }
70  }
71 
72  public:
78  WCycle(std::shared_ptr<Level<D>> finest_level, const CycleOpts &opts) : Cycle<D>(finest_level)
79  {
