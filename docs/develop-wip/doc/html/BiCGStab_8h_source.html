49  static int solve(std::shared_ptr<VectorGenerator<D>> vg, std::shared_ptr<const Operator<D>> A,
50  std::shared_ptr<Vector<D>> x, std::shared_ptr<const Vector<D>> b,
51  std::shared_ptr<const Operator<D>> Mr = nullptr, int max_it = 1000,
52  double tolerance = 1e-12, std::shared_ptr<ThunderEgg::Timer> timer = nullptr)
53  {
54  std::shared_ptr<Vector<D>> resid = vg->getNewVector();
55  std::shared_ptr<Vector<D>> ms;
56  std::shared_ptr<Vector<D>> mp;
57  if (Mr != nullptr) {
58  ms = vg->getNewVector();
59  mp = vg->getNewVector();
60  }
61  A->apply(x, resid);
62  resid->scaleThenAdd(-1, b);
63  double r0_norm = resid->twoNorm();
64  std::shared_ptr<Vector<D>> rhat = vg->getNewVector();
65  rhat->copy(resid);
66  std::shared_ptr<Vector<D>> p = vg->getNewVector();
67  p->copy(resid);
68  std::shared_ptr<Vector<D>> ap = vg->getNewVector();
69  std::shared_ptr<Vector<D>> as = vg->getNewVector();
70 
71  std::shared_ptr<Vector<D>> s = vg->getNewVector();
72  double rho = rhat->dot(resid);
73 
74  int num_its = 0;
75  if (r0_norm == 0) {
76  return num_its;
77  }
78  double residual = resid->twoNorm() / r0_norm;
79  while (residual > tolerance && num_its < max_it) {
80  if (timer) {
81  timer->start("Iteration");
82  }
83 
84  if (Mr != nullptr) {
85  /* Solve M*mp = p; */
86  printf("%5d %16.8e\n", num_its, residual);
87  Mr->apply(p, mp);
88  A->apply(mp, ap);
89  } else {
90  A->apply(p, ap);
91  // printf(" %5d %16.8e\n", num_its, residual);
92  }
93  double alpha = rho / rhat->dot(ap);
94  s->copy(resid);
95  s->addScaled(-alpha, ap);
96  if (s->twoNorm() / r0_norm <= tolerance) {
97  x->addScaled(alpha, p);
98  if (timer) {
99  timer->stop("Iteration");
100  }
101  break;
102  }
103  if (Mr != nullptr) {
104  Mr->apply(s, ms);
105  A->apply(ms, as);
106  } else {
107  A->apply(s, as);
108  }
109  double omega = as->dot(s) / as->dot(as);
110 
111  // update x and residual
112  if (Mr != nullptr) {
113  x->addScaled(alpha, mp, omega, ms);
114  } else {
115  x->addScaled(alpha, p, omega, s);
116  }
117  resid->addScaled(-alpha, ap, -omega, as);
118 
119  double rho_new = resid->dot(rhat);
120  double beta = rho_new * alpha / (rho * omega);
121  p->addScaled(-omega, ap);
122  p->scaleThenAdd(beta, resid);
123 
124  num_its++;
125  rho = rho_new;
126  residual = resid->twoNorm() / r0_norm;
127  // printf("%5d %16.8e\n", num_its, residual);
128 
129  if (timer) {
130  timer->stop("Iteration");
131  }
132  }
133  return num_its;
134  }
