71  void applySinglePatch(std::shared_ptr<const PatchInfo<D>> pinfo,
72  const std::vector<LocalData<D>> & us,
73  std::vector<LocalData<D>> & fs) const override
74  {
75  const LocalData<D> c = coeffs->getLocalData(0, pinfo->local_index);
76  std::array<double, D> h2 = pinfo->spacings;
77  for (size_t i = 0; i < D; i++) {
78  h2[i] *= h2[i];
79  }
80  loop<0, D - 1>([&](int axis) {
81  Side<D> lower_side(axis * 2);
82  Side<D> upper_side(axis * 2 + 1);
83  if (!pinfo->hasNbr(lower_side)) {
84  LocalData<D - 1> lower = us[0].getGhostSliceOnSide(lower_side, 1);
85  const LocalData<D - 1> mid = us[0].getSliceOnSide(lower_side);
86  nested_loop<D - 1>(mid.getStart(), mid.getEnd(), [&](std::array<int, D - 1> coord) {
87  lower[coord] = -mid[coord];
88  });
89  }
90  if (!pinfo->hasNbr(upper_side)) {
91  LocalData<D - 1> upper = us[0].getGhostSliceOnSide(upper_side, 1);
92  const LocalData<D - 1> mid = us[0].getSliceOnSide(upper_side);
93  nested_loop<D - 1>(mid.getStart(), mid.getEnd(), [&](std::array<int, D - 1> coord) {
94  upper[coord] = -mid[coord];
95  });
96  }
97  int stride = us[0].getStrides()[axis];
98  int c_stride = c.getStrides()[axis];
99  nested_loop<D>(us[0].getStart(), us[0].getEnd(), [&](std::array<int, D> coord) {
100  const double *ptr = us[0].getPtr(coord);
101  const double *c_ptr = c.getPtr(coord);
102  double lower = *(ptr - stride);
103  double mid = *ptr;
104  double upper = *(ptr + stride);
105  double c_lower = *(c_ptr - c_stride);
106  double c_mid = *c_ptr;
107  double c_upper = *(c_ptr + c_stride);
108  fs[0][coord]
109  = addValue(axis) * fs[0][coord]
110  + ((c_upper + c_mid) * (upper - mid) - (c_lower + c_mid) * (mid - lower))
111  / (2 * h2[axis]);
112  });
113  });
114  }
115  void addGhostToRHS(std::shared_ptr<const PatchInfo<D>> pinfo,
116  const std::vector<LocalData<D>> & us,
117  std::vector<LocalData<D>> & fs) const override
118  {
119  const LocalData<D> c = coeffs->getLocalData(0, pinfo->local_index);
120  for (Side<D> s : Side<D>::getValues()) {
121  if (pinfo->hasNbr(s)) {
122  double h2 = pow(pinfo->spacings[s.getAxisIndex()], 2);
123  LocalData<D - 1> f_inner = fs[0].getSliceOnSide(s);
124  LocalData<D - 1> u_ghost = us[0].getSliceOnSide(s, -1);
125  const LocalData<D - 1> u_inner = us[0].getSliceOnSide(s);
126  const LocalData<D - 1> c_ghost = c.getSliceOnSide(s, -1);
127  const LocalData<D - 1> c_inner = c.getSliceOnSide(s);
128  nested_loop<D - 1>(
129  f_inner.getStart(), f_inner.getEnd(), [&](const std::array<int, D - 1> &coord) {
130  f_inner[coord] -= (u_ghost[coord] + u_inner[coord])
131  * (c_inner[coord] + c_ghost[coord]) / (2 * h2);
132  u_ghost[coord] = 0;
133  });
134  }
135  }
136  }
144  void addDrichletBCToRHS(std::shared_ptr<Vector<D>> f,
145  std::function<double(const std::array<double, D> &)> gfunc,
146  std::function<double(const std::array<double, D> &)> hfunc)
147  {
148  for (int i = 0; i < f->getNumLocalPatches(); i++) {
149  LocalData<D> f_ld = f->getLocalData(0, i);
150  auto pinfo = this->domain->getPatchInfoVector()[i];
151  for (Side<D> s : Side<D>::getValues()) {
152  if (!pinfo->hasNbr(s)) {
153  double h2 = pow(pinfo->spacings[s.getAxisIndex()], 2);
154  LocalData<D - 1> ld = f_ld.getSliceOnSide(s);
155  nested_loop<D - 1>(
156  ld.getStart(), ld.getEnd(), [&](const std::array<int, D - 1> &coord) {
157  std::array<double, D> real_coord;
158  DomainTools<D>::getRealCoordBound(pinfo, coord, s, real_coord);
159  std::array<double, D> other_real_coord = real_coord;
160  if (s.isLowerOnAxis()) {
161  other_real_coord[s.getAxisIndex()] -= pinfo->spacings[s.getAxisIndex()];
162  } else {
163  other_real_coord[s.getAxisIndex()] += pinfo->spacings[s.getAxisIndex()];
164  }
165  ld[coord] -= 2 * gfunc(real_coord) * hfunc(real_coord) / h2;
166  });
167  }
168  }
169  }
170  }
