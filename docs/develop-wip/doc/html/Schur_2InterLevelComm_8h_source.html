113 inline InterLevelComm<D>::InterLevelComm(std::shared_ptr<SchurHelper<D>> coarse_sh,
114  std::shared_ptr<SchurHelper<D>> fine_sh)
115 {
116  using namespace std;
117  n = coarse_sh->getN();
118  set<int> parent_ids;
119  for (auto &p : fine_sh->getIfaces()) {
120  IfaceSet<D> &ifs = *p.second;
121  if(ifs.parent_id!=-1){
122  parent_ids.insert(d.parent_id);
123  }
124  }
125  vector<int> coarse_parent_global_index_map_vec(parent_ids.begin(), parent_ids.end());
126  vector<int> coarse_parent_gid_map_vec = coarse_parent_global_index_map_vec;
127  // get global indexes for parent domains
128  PW<AO> ao;
129  AOCreateMapping(MPI_COMM_WORLD, coarse_dc->domain_gid_map_vec.size(),
130  &coarse_dc->domain_gid_map_vec[0], &coarse_dc->domain_map_vec[0], &ao);
131  AOApplicationToPetsc(ao, coarse_parent_global_index_map_vec.size(),
132  &coarse_parent_global_index_map_vec[0]);
133 
134  // set index info
135  map<int, int> gid_to_local;
136  map<int, int> gid_to_global;
137 
138  for (size_t i = 0; i < coarse_parent_gid_map_vec.size(); i++) {
139  int gid = coarse_parent_gid_map_vec[i];
140  gid_to_local[gid] = i;
141  gid_to_global[gid] = coarse_parent_global_index_map_vec[i];
142  }
143  for (auto &p : fine_dc->domains) {
144  Domain<D> & d = *p.second;
145  int gid = d.parent_id;
146  ILCFineToCoarseMetadata<D> data = {p.second, gid_to_local[gid], gid_to_global[gid]};
147  coarse_domains.insert(data);
148  }
149 
150  PW<IS> dist_is;
151  ISCreateBlock(MPI_COMM_SELF, (int) pow(n, D-1), coarse_parent_global_index_map_vec.size(),
152  &coarse_parent_global_index_map_vec[0], PETSC_COPY_VALUES, &dist_is);
153 
154  local_vec_size = coarse_parent_global_index_map_vec.size() * pow(n, D-1);
155 
156  PW<Vec> u_local = getNewCoarseDistVec();
157  PW<Vec> u = coarse_dc->getNewDomainVec();
158  VecScatterCreate(u, dist_is, u_local, nullptr, &scatter);
159 }
