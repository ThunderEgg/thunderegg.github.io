140 template <size_t D> inline void DomGen<D>::extractLevel()
141 {
142  int rank;
143  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
144  PInfoMap new_level;
145  if (rank == 0) {
146  Node<D> child = *t.levels.at(curr_level);
147  std::deque<int> q;
148  std::set<int> qed;
149  q.push_back(child.id);
150  qed.insert(child.id);
151 
152  while (!q.empty()) {
153  std::shared_ptr<PatchInfo<D>> d_ptr(new PatchInfo<D>());
154  PatchInfo<D> & pinfo = *d_ptr;
155  pinfo.num_ghost_cells = num_ghost_cells;
156  Node<D> n = t.nodes.at(q.front());
157  q.pop_front();
158 
159  pinfo.ns = ns;
160  pinfo.id = n.id;
161  for (size_t i = 0; i < D; i++) {
162  pinfo.spacings[i] = n.lengths[i] / ns[i];
163  }
164  pinfo.starts = n.starts;
165  pinfo.refine_level = n.level;
166  if (n.level < curr_level) {
167  // will still exist at next coarsening
168  pinfo.parent_id = n.id;
169  if (curr_level != num_levels) {
170  pinfo.child_ids[0] = n.id;
171  }
172  } else {
173  pinfo.parent_id = n.parent;
174  if (pinfo.parent_id != -1) {
175  auto orth_iter = Orthant<D>::getValues().begin();
176  while (t.nodes.at(n.parent).child_id[orth_iter->getIndex()] != n.id) {
177  ++orth_iter;
178  }
179  pinfo.orth_on_parent = *orth_iter;
180  }
181  if (!n.hasChildren() && curr_level != num_levels) {
182  pinfo.child_ids[0] = n.id;
183  } else if (n.hasChildren()) {
184  pinfo.child_ids = n.child_id;
185  }
186  }
187  // set child ranks
188  if (pinfo.child_ids[0] != -1) {
189  for (size_t i = 0; i < pinfo.child_ids.size(); i++) {
190  if (pinfo.child_ids[i] != -1) {
191  pinfo.child_ranks[i] = id_rank_map.at(pinfo.child_ids[i]);
192  }
193  }
194  }
195 
196  // set and enqueue nbrs
197  for (Side<D> s : Side<D>::getValues()) {
198  if (n.nbrId(s) == -1 && n.parent != -1 && t.nodes.at(n.parent).nbrId(s) != -1) {
199  Node<D> parent = t.nodes.at(n.parent);
200  Node<D> nbr = t.nodes.at(parent.nbrId(s));
201  auto octs = Orthant<D>::getValuesOnSide(s);
202  auto orth_iter = Orthant<D - 1>::getValues().begin();
203  while (parent.childId(octs[orth_iter->getIndex()]) != n.id) {
204  ++orth_iter;
205  }
206  pinfo.nbr_info[s.getIndex()].reset(new CoarseNbrInfo<D>(nbr.id, *orth_iter));
207  if (!qed.count(nbr.id)) {
208  q.push_back(nbr.id);
209  qed.insert(nbr.id);
210  }
211  } else if (n.level < curr_level && n.nbrId(s) != -1
212  && t.nodes.at(n.nbrId(s)).hasChildren()) {
213  Node<D> nbr = t.nodes.at(n.nbrId(s));
214  auto octs = Orthant<D>::getValuesOnSide(s.opposite());
215  std::array<int, Orthant<D>::num_orthants / 2> nbr_ids;
216  for (size_t i = 0; i < Orthant<D>::num_orthants / 2; i++) {
217  int id = nbr.childId(octs[i]);
218  nbr_ids[i] = id;
219  if (!qed.count(id)) {
220  q.push_back(id);
221  qed.insert(id);
222  }
223  }
224  pinfo.nbr_info[s.getIndex()].reset(new FineNbrInfo<D>(nbr_ids));
225  } else if (n.nbrId(s) != -1) {
226  int id = n.nbrId(s);
227  if (!qed.count(id)) {
228  q.push_back(id);
229  qed.insert(id);
230  }
231  pinfo.nbr_info[s.getIndex()].reset(new NormalNbrInfo<D>(id));
232  }
233  }
234  new_level[pinfo.id] = d_ptr;
235  }
236  for (auto &p : new_level) {
237  p.second->setPtrs(new_level);
238  }
239  }
240  // balance
241  if (curr_level == num_levels) {
242  balanceLevel(new_level);
243  } else {
244  auto &old_level = domain_list.back()->getPatchInfoMap();
245  balanceLevelWithLower(new_level, old_level);
246  // update parent ranks
247  std::map<int, int> id_rank_map;
248  // get outgoing information
249  std::map<int, std::set<int>> out_info;
250  for (auto pair : new_level) {
251  id_rank_map[pair.first] = rank;
252  auto &pinfo = pair.second;
253  for (size_t i = 0; i < Orthant<D>::num_orthants; i++) {
254  if (pinfo->child_ranks[i] != -1 && pinfo->child_ranks[i] != rank) {
255  out_info[pinfo->child_ranks[i]].insert(pinfo->id);
256  }
257  }
258  }
259  // get incoming information
260  std::set<int> incoming_ids;
261  for (auto &pair : old_level) {
262  if (!new_level.count(pair.second->parent_id)) {
263  incoming_ids.insert(pair.second->parent_id);
264  }
265  }
266  // send info
267  std::deque<std::vector<int>> buffers;
268  std::vector<MPI_Request> send_requests;
269  for (auto &pair : out_info) {
270  int dest = pair.first;
271  std::vector<int> buffer(pair.second.begin(), pair.second.end());
272  buffers.push_back(buffer);
273  MPI_Request request;
274  MPI_Isend(buffer.data(), buffer.size(), MPI_INT, dest, 0, MPI_COMM_WORLD, &request);
275  send_requests.push_back(request);
276  }
277  // recv info
278  while (incoming_ids.size()) {
279  MPI_Status status;
280  MPI_Probe(MPI_ANY_SOURCE, 0, MPI_COMM_WORLD, &status);
281  int size;
282  MPI_Get_count(&status, MPI_INT, &size);
283  int *buffer = new int[size];
284 
285  MPI_Recv(buffer, size, MPI_INT, status.MPI_SOURCE, 0, MPI_COMM_WORLD, &status);
286 
287  for (int i = 0; i < size; i++) {
288  id_rank_map[buffer[i]] = status.MPI_SOURCE;
289  incoming_ids.erase(buffer[i]);
290  }
291 
292  delete[] buffer;
293  }
294  // wait for all
295  MPI_Waitall(send_requests.size(), &send_requests[0], MPI_STATUSES_IGNORE);
296  // update rank info
297  for (auto &pair : old_level) {
298  pair.second->parent_rank = id_rank_map.at(pair.second->parent_id);
299  }
300  }
301  domain_list.push_back(
302  std::shared_ptr<Domain<D>>(new Domain<D>(new_level, ns, num_ghost_cells)));
303  if (neumann) {
304  IsNeumannFunc<D> inf = [](Side<D>, const std::array<double, D> &,
305  const std::array<double, D> &) { return true; };
306  domain_list.back()->setNeumann(inf);
307  }
308  curr_level--;
309 }
310 template <size_t D> inline void DomGen<D>::balanceLevel(PInfoMap &level)
311 {
312  struct Zoltan_Struct *zz = Zoltan_Create(MPI_COMM_WORLD);
313 
314  // parameters
315  Zoltan_Set_Param(zz, "LB_METHOD", "HYPERGRAPH"); /* Zoltan method: HYPERGRAPH */
316  Zoltan_Set_Param(zz, "LB_APPROACH", "PARTITION"); /* Zoltan method: "BLOCK" */
317  Zoltan_Set_Param(zz, "NUM_GID_ENTRIES", "1"); /* global ID is 1 integer */
318  Zoltan_Set_Param(zz, "NUM_LID_ENTRIES", "0"); /* don't use local IDs */
319  Zoltan_Set_Param(zz, "OBJ_WEIGHT_DIM", "0"); /* we omit object weights */
320  Zoltan_Set_Param(zz, "AUTO_MIGRATE", "FALSE"); /* we omit object weights */
321  Zoltan_Set_Param(zz, "DEBUG_LEVEL", "0"); /* we omit object weights */
322  Zoltan_Set_Param(zz, "RETURN_LISTS", "PARTS"); /* we omit object weights */
323 
324  // Query functions
325  // Number of Vertices
326  auto numObjFn = [](void *data, int *ierr) -> int {
327  PInfoMap &map = *(PInfoMap *) data;
328  *ierr = ZOLTAN_OK;
329  return map.size();
330  };
331  Zoltan_Set_Num_Obj_Fn(zz, numObjFn, &level);
332 
333  // List of vertices
334  auto objListFn
335  = [](void *data, int num_gid_entries, int num_lid_entries, ZOLTAN_ID_PTR global_ids,
336  ZOLTAN_ID_PTR local_ids, int wgt_dim, float *obj_wgts, int *ierr) {
337  PInfoMap &map = *(PInfoMap *) data;
338  *ierr = ZOLTAN_OK;
339  int pos = 0;
340  for (auto p : map) {
341  global_ids[pos] = p.first;
342  pos++;
343  }
344  };
345  Zoltan_Set_Obj_List_Fn(zz, objListFn, &level);
346 
347  // Construct hypergraph
348  struct CompressedVertex {
349  std::vector<int> vertices;
350  std::vector<int> ptrs;
351  std::vector<int> edges;
352  };
353  CompressedVertex graph;
354  for (auto &p : level) {
355  PatchInfo<D> &pinfo = *p.second;
356  graph.vertices.push_back(pinfo.id);
357  graph.ptrs.push_back(graph.edges.size());
358  for (Side<D> s : Side<D>::getValues()) {
359  int edge_id = -1;
360  if (pinfo.hasNbr(s)) {
361  switch (pinfo.getNbrType(s)) {
362  case NbrType::Normal:
363  if (s.isLowerOnAxis()) {
364  edge_id = pinfo.getNormalNbrInfo(s).id;
365  } else {
366  edge_id = pinfo.id;
367  }
368  break;
369  case NbrType::Fine:
370  edge_id = pinfo.id;
371  break;
372  case NbrType::Coarse:
373  edge_id = pinfo.getCoarseNbrInfo(s).id;
374  break;
375  }
376  }
377  graph.edges.push_back(edge_id * Side<D>::num_sides + s.getIndex());
378  }
379  }
380 
381  // set graph functions
382  Zoltan_Set_HG_Size_CS_Fn(zz,
383  [](void *data, int *num_lists, int *num_pins, int *format, int *ierr) {
384  CompressedVertex &graph = *(CompressedVertex *) data;
385  *ierr = ZOLTAN_OK;
386  *num_lists = graph.vertices.size();
387  *num_pins = graph.edges.size();
388  *format = ZOLTAN_COMPRESSED_VERTEX;
389  },
390  &graph);
391  Zoltan_Set_HG_CS_Fn(zz,
392  [](void *data, int num_gid_entries, int num_vtx_edge, int num_pins,
393  int format, ZOLTAN_ID_PTR vtxedge_GID, int *vtxedge_ptr,
394  ZOLTAN_ID_PTR pin_GID, int *ierr) {
395  CompressedVertex &graph = *(CompressedVertex *) data;
396  *ierr = ZOLTAN_OK;
397  copy(graph.vertices.begin(), graph.vertices.end(), vtxedge_GID);
398  copy(graph.ptrs.begin(), graph.ptrs.end(), vtxedge_ptr);
399  copy(graph.edges.begin(), graph.edges.end(), pin_GID);
400  },
401  &graph);
402 
403  Zoltan_Set_Obj_Size_Fn(zz,
404  [](void *data, int num_gid_entries, int num_lid_entries,
405  ZOLTAN_ID_PTR global_id, ZOLTAN_ID_PTR local_id, int *ierr) {
406  PInfoMap &map = *(PInfoMap *) data;
407  *ierr = ZOLTAN_OK;
408  return map[*global_id]->serialize(nullptr);
409  },
410  &level);
411  Zoltan_Set_Pack_Obj_Fn(zz,
412  [](void *data, int num_gid_entries, int num_lid_entries,
413  ZOLTAN_ID_PTR global_id, ZOLTAN_ID_PTR local_id, int dest, int size,
414  char *buf, int *ierr) {
415  PInfoMap &map = *(PInfoMap *) data;
416  *ierr = ZOLTAN_OK;
417  map[*global_id]->serialize(buf);
418  map.erase(*global_id);
419  },
420  &level);
421  Zoltan_Set_Unpack_Obj_Fn(
422  zz,
423  [](void *data, int num_gid_entries, ZOLTAN_ID_PTR global_id, int size, char *buf, int *ierr) {
424  PInfoMap &map = *(PInfoMap *) data;
425  *ierr = ZOLTAN_OK;
426  map[*global_id].reset(new PatchInfo<D>());
427  map[*global_id]->deserialize(buf);
428  },
429  &level);
430  // Zoltan_Set_Obj_Size_Multi_Fn(zz, DomainZoltanHelper::object_sizes, this);
431  // Zoltan_Set_Pack_Obj_Multi_Fn(zz, DomainZoltanHelper::pack_objects, this);
432  // Zoltan_Set_Unpack_Obj_Multi_Fn(zz, DomainZoltanHelper::unpack_objects, this);
433 
435  // Zoltan can now partition the objects in this collection.
436  // In this simple example, we assume the number of partitions is
437  // equal to the number of processes. Process rank 0 will own
438  // partition 0, process rank 1 will own partition 1, and so on.
440 
441  int changes;
442  int numGidEntries;
443  int numLidEntries;
444  int numImport;
445  ZOLTAN_ID_PTR importGlobalIds;
446  ZOLTAN_ID_PTR importLocalIds;
447  int * importProcs;
448  int * importToPart;
449  int numExport;
450  ZOLTAN_ID_PTR exportGlobalIds;
451  ZOLTAN_ID_PTR exportLocalIds;
452  int * exportProcs;
453  int * exportToPart;
454 
455  int rc = Zoltan_LB_Partition(zz, &changes, &numGidEntries, &numLidEntries, &numImport,
456  &importGlobalIds, &importLocalIds, &importProcs, &importToPart,
457  &numExport, &exportGlobalIds, &exportLocalIds, &exportProcs,
458  &exportToPart);
459 
460  // update ranks of neighbors before migrating
461  id_rank_map.clear();
462  for (int i = 0; i < numExport; i++) {
463  int curr_id = exportGlobalIds[i];
464  int dest_rank = exportProcs[i];
465  id_rank_map[curr_id] = dest_rank;
466  level[curr_id]->updateRank(dest_rank);
467  }
468 
469  rc = Zoltan_Migrate(zz, numImport, importGlobalIds, importLocalIds, importProcs, importToPart,
470  numExport, exportGlobalIds, exportLocalIds, exportProcs, exportToPart);
471 
472  if (rc != ZOLTAN_OK) {
473  throw "zoltan error";
474  }
475  Zoltan_Destroy(&zz);
476 #if DD_DEBUG
477  std::cout << "I have " << level.size() << " domains: ";
478 
479  int prev = -100;
480  bool range = false;
481  for (auto &p : level) {
482  int curr = p.second->id;
483  if (curr != prev + 1 && !range) {
484  std::cout << curr << "-";
485  range = true;
486  } else if (curr != prev + 1 && range) {
487  std::cout << prev << " " << curr << "-";
488  }
489  prev = curr;
490  }
491 
492  std::cout << prev << "\n";
493  std::cout << std::endl;
494 #endif
495  for (auto &p : level) {
496  p.second->setPtrs(level);
497  }
498 }
499 template <size_t D>
500 inline void DomGen<D>::balanceLevelWithLower(PInfoMap &level, PInfoMap &lower_level)
501 {
502  struct Zoltan_Struct *zz = Zoltan_Create(MPI_COMM_WORLD);
503 
504  // parameters
505  Zoltan_Set_Param(zz, "LB_METHOD", "HYPERGRAPH"); /* Zoltan method: HYPERGRAPH */
506  Zoltan_Set_Param(zz, "LB_APPROACH", "PARTITION"); /* Zoltan method: "BLOCK" */
507  Zoltan_Set_Param(zz, "NUM_GID_ENTRIES", "1"); /* global ID is 1 integer */
508  Zoltan_Set_Param(zz, "NUM_LID_ENTRIES", "0"); /* don't use local IDs */
509  Zoltan_Set_Param(zz, "OBJ_WEIGHT_DIM", "1"); /* we omit object weights */
510  Zoltan_Set_Param(zz, "EDGE_WEIGHT_DIM", "0"); /* we omit object weights */
511  Zoltan_Set_Param(zz, "AUTO_MIGRATE", "FALSE"); /* we omit object weights */
512  Zoltan_Set_Param(zz, "DEBUG_LEVEL", "0"); /* we omit object weights */
513  Zoltan_Set_Param(zz, "RETURN_LISTS", "PARTS"); /* we omit object weights */
514 
515  // Query functions
516  // Number of Vertices
517  struct Levels {
518  PInfoMap *upper;
519  PInfoMap *lower;
520  };
521  Levels levels = {&level, &lower_level};
522  Zoltan_Set_Num_Obj_Fn(zz,
523  [](void *data, int *ierr) -> int {
524  Levels *levels = (Levels *) data;
525  *ierr = ZOLTAN_OK;
526  return levels->upper->size() + levels->lower->size();
527  },
528  &levels);
529 
530  // List of vertices
531  Zoltan_Set_Obj_List_Fn(zz,
532  [](void *data, int num_gid_entries, int num_lid_entries,
533  ZOLTAN_ID_PTR global_ids, ZOLTAN_ID_PTR local_ids, int wgt_dim,
534  float *obj_wgts, int *ierr) {
535  Levels *levels = (Levels *) data;
536  *ierr = ZOLTAN_OK;
537  int pos = 0;
538  for (auto p : *levels->upper) {
539  global_ids[pos] = p.first;
540  obj_wgts[pos] = 1;
541  pos++;
542  }
543  for (auto p : *levels->lower) {
544  global_ids[pos] = p.first + levels->upper->size();
545  obj_wgts[pos] = 0;
546  pos++;
547  }
548  },
549  &levels);
550 
551  // Construct hypergraph
552  struct CompressedVertex {
553  std::vector<int> vertices;
554  std::vector<int> ptrs;
555  std::vector<int> edges;
556  };
557  CompressedVertex graph;
558  // process coarse level
559  for (auto &p : level) {
560  PatchInfo<D> &pinfo = *p.second;
561  graph.vertices.push_back(pinfo.id);
562  graph.ptrs.push_back(graph.edges.size());
563  // patch to patch communication
564  for (Side<D> s : Side<D>::getValues()) {
565  int edge_id = -1;
566  if (pinfo.hasNbr(s)) {
567  switch (pinfo.getNbrType(s)) {
568  case NbrType::Normal:
569  if (s.isLowerOnAxis()) {
570  edge_id = pinfo.getNormalNbrInfo(s).id;
571  } else {
572  edge_id = pinfo.id;
573  }
574  break;
575  case NbrType::Fine:
576  edge_id = pinfo.id;
577  break;
578  case NbrType::Coarse:
579  edge_id = pinfo.getCoarseNbrInfo(s).id;
580  break;
581  }
582  }
583  graph.edges.push_back(edge_id * Side<D>::num_sides + s.getIndex());
584  }
585  // level to level communication
586  graph.edges.push_back(-pinfo.id - 1);
587  }
588  // process fine level
589  for (auto &p : lower_level) {
590  PatchInfo<D> &pinfo = *p.second;
591  graph.vertices.push_back(pinfo.id + level.size());
592  graph.ptrs.push_back(graph.edges.size());
593  graph.edges.push_back(-pinfo.parent_id - 1);
594  }
595 
596  // set graph functions
597  Zoltan_Set_HG_Size_CS_Fn(zz,
598  [](void *data, int *num_lists, int *num_pins, int *format, int *ierr) {
599  CompressedVertex &graph = *(CompressedVertex *) data;
600  *ierr = ZOLTAN_OK;
601  *num_lists = graph.vertices.size();
602  *num_pins = graph.edges.size();
603  *format = ZOLTAN_COMPRESSED_VERTEX;
604  },
605  &graph);
606  Zoltan_Set_HG_CS_Fn(zz,
607  [](void *data, int num_gid_entries, int num_vtx_edge, int num_pins,
608  int format, ZOLTAN_ID_PTR vtxedge_GID, int *vtxedge_ptr,
609  ZOLTAN_ID_PTR pin_GID, int *ierr) {
610  CompressedVertex &graph = *(CompressedVertex *) data;
611  *ierr = ZOLTAN_OK;
612  copy(graph.vertices.begin(), graph.vertices.end(), vtxedge_GID);
613  copy(graph.ptrs.begin(), graph.ptrs.end(), vtxedge_ptr);
614  copy(graph.edges.begin(), graph.edges.end(), pin_GID);
615  },
616  &graph);
617 
618  Zoltan_Set_Obj_Size_Fn(zz,
619  [](void *data, int num_gid_entries, int num_lid_entries,
620  ZOLTAN_ID_PTR global_id, ZOLTAN_ID_PTR local_id, int *ierr) {
