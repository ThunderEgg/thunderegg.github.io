45  static int GetRank(std::shared_ptr<PatchInfo<D>> pinfo, Side<D> s)
46  {
47  if (s.isLowerOnAxis()) {
48  // lower axis interface belongs to neighboring rank
49  auto nbr_info = pinfo->getNormalNbrInfoPtr(s);
50  return nbr_info->rank;
51  } else {
52  // higher axis interafce belongs to this patch's rank
53  return pinfo->rank;
54  }
55  }
56 
64  static int GetId(std::shared_ptr<PatchInfo<D>> pinfo, Side<D> s)
65  {
66  if (s.isLowerOnAxis()) {
67  // lower axis interface belongs to neighboring rank
68  auto nbr_info = pinfo->getNormalNbrInfoPtr(s);
69  return (int) (nbr_info->id * Side<D>::num_sides + s.opposite().getIndex());
70  } else {
71  // higher axis interafce belongs to this patch's rank
72  return (int) (pinfo->id * Side<D>::num_sides + s.getIndex());
73  }
74  }
