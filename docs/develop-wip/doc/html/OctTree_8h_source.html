96 template <size_t D> inline Tree<D>::Tree(std::string file_name)
97 {
98  using namespace std;
99  ifstream input(file_name, ios_base::binary);
100  int num_nodes;
101  int num_trees;
102  input.read((char *) &num_nodes, 4);
103  input.read((char *) &num_trees, 4);
104  num_levels = 0;
105  max_id = 0;
106  for (int i = 0; i < num_nodes; i++) {
107  Node<D> n;
108  // id level parent
109  input.read((char *) &n.id, 3 * 4);
110  // length and starts
111  input.read((char *) &n.lengths[0], sizeof(n.lengths));
112  input.read((char *) &n.starts[0], sizeof(n.starts));
113 
114  input.read((char *) &n.nbr_id[0], sizeof(n.nbr_id));
115  input.read((char *) &n.child_id[0], sizeof(n.child_id));
116 
117  if (i == 0) {
118  root = n.id;
119  }
120  max_id = max(max_id, n.id);
121  nodes[n.id] = n;
122 
123  if (n.level > num_levels) {
124  num_levels = n.level;
125  }
126  levels[n.level] = &nodes[n.id];
127  }

129 template <size_t D> inline void Tree<D>::refineLeaves()
130 {
131  using namespace std;
132  Node<D> root_node = nodes[root];
133  Node<D> child = root_node;
134  int level = 0;
135  while (child.hasChildren()) {
136  child = nodes[child.child_id[0]];
137  level++;
138  }
139  deque<pair<int, int>> q;
140  set<pair<int, int>> qed;
141  q.push_back(make_pair(level, child.id));
142  qed.insert(make_pair(level, child.id));
143 
144  while (!q.empty()) {
145  pair<int, int> p = q.front();
146  Node<D> n = nodes[p.second];
147  int level = p.first;
148  q.pop_front();
149 
150  // set and enqueue nbrs
151  for (Side<D> s : Side<D>::getValues()) {
152  // coarser
153  if (n.nbrId(s) == -1 && n.parent != -1 && nodes[n.parent].nbrId(s) != -1) {
154  Node<D> parent = nodes[n.parent];
155  Node<D> nbr = nodes[parent.nbrId(s)];
156  pair<int, int> p(level - 1, nbr.id);
157  if (!qed.count(p)) {
158  q.push_back(p);
159  qed.insert(p);
160  }
161  // finer
162  } else if (n.nbrId(s) != -1 && nodes[n.nbrId(s)].hasChildren()) {
163  Node<D> nbr = nodes[n.nbrId(s)];
164  auto octs = Orthant<D>::getValuesOnSide(s.opposite());
165  for (size_t i = 0; i < octs.size(); i++) {
166  int id = nbr.childId(octs[i]);
167  pair<int, int> p(level + 1, id);
168  if (!qed.count(p)) {
169  q.push_back(p);
170  qed.insert(p);
171  }
172  }
173  // normal
174  } else if (n.nbrId(s) != -1) {
175  int id = n.nbrId(s);
176  pair<int, int> p(level, id);
177  if (!qed.count(p)) {
178  q.push_back(p);
179  qed.insert(p);
180  }
181  }
182  }
183  }
184  for (auto p : qed) {
185  refineNode(nodes[p.second]);
186  }
187  levels[num_levels + 1] = &nodes.at(levels[num_levels]->child_id[0]);
188  this->num_levels++;
189 }
190 template <size_t D> inline void Tree<D>::refineNode(Node<D> &n)
191 {
192  std::array<Node<D>, Orthant<D>::num_orthants> new_children;
193  for (Orthant<D> o : Orthant<D>::getValues()) {
194  new_children[o.getIndex()] = Node<D>(n, o);
195  Node<D> &child = new_children[o.getIndex()];
196  max_id++;
197  child.id = max_id;
198  n.child_id[o.getIndex()] = max_id;
199  }
200  // set new neighbors
201  for (Orthant<D> o : Orthant<D>::getValues()) {
202  for (Side<D> s : o.getInteriorSides()) {
203  new_children[o.getIndex()].nbrId(s) = new_children[o.getNbrOnSide(s).getIndex()].id;
204  }
205  }
206 
207  // set outer neighbors
208  for (Side<D> s : Side<D>::getValues()) {
209  if (n.hasNbr(s) && nodes[n.nbrId(s)].hasChildren()) {
210  Node<D> &nbr = nodes[n.nbrId(s)];
211  for (Orthant<D> o : Orthant<D>::getValuesOnSide(s)) {
212  Node<D> &child = new_children[o.getIndex()];
213  Node<D> &nbr_child = nodes[nbr.childId(o.getNbrOnSide(s))];
214  child.nbrId(s) = nbr_child.id;
215  nbr_child.nbrId(s.opposite()) = child.id;
216  }
217  }
218  }
219  // add nodes
220  for (Node<D> child : new_children) {
221  nodes[child.id] = child;
222  }
223 }
