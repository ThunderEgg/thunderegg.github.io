66  static int applyMat(Mat A, Vec x, Vec b)
67  {
68  MatShellCreator<D> *msc = nullptr;
69  MatShellGetContext(A, &msc);
70 
71  auto te_x = msc->vg->getNewVector();
72  auto te_b = msc->vg->getNewVector();
73 
74  // petsc vectors don't have gost padding for patchs, so this is neccesary
75  const double *x_view;
76  VecGetArrayRead(x, &x_view);
77  int index = 0;
78  for (int p_index = 0; p_index < te_x->getNumLocalPatches(); p_index++) {
79  for (int c = 0; c < te_x->getNumComponents(); c++) {
80  LocalData<D> ld = te_x->getLocalData(c, p_index);
81  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
82  ld[coord] = x_view[index];
83  index++;
84  });
85  }
86  }
87 
88  VecRestoreArrayRead(x, &x_view);
89 
90  msc->op->apply(te_x, te_b);
91 
92  double *b_view;
93  VecGetArray(b, &b_view);
94  index = 0;
95  for (int p_index = 0; p_index < te_b->getNumLocalPatches(); p_index++) {
96  for (int c = 0; c < te_b->getNumComponents(); c++) {
97  const LocalData<D> ld = te_b->getLocalData(c, p_index);
98  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
99  b_view[index] = ld[coord];
100  index++;
101  });
102  }
103  }
104  VecRestoreArray(b, &b_view);
105 
106  return 0;
107  }

130  static Mat GetNewMatShell(std::shared_ptr<Operator<D>> op,
131  std::shared_ptr<VectorGenerator<D>> vg)
132  {
133  MatShellCreator<D> *msc = new MatShellCreator(op, vg);
134  auto vec = vg->getNewVector();
135  int m = vec->getNumLocalCells() * vec->getNumComponents();
136  Mat A;
137  MatCreateShell(MPI_COMM_WORLD, m, m, PETSC_DETERMINE, PETSC_DETERMINE, msc, &A);
138  MatShellSetOperation(A, MATOP_MULT, (void (*)(void)) applyMat);
139  MatShellSetOperation(A, MATOP_DESTROY, (void (*)(void)) destroyMat);
140  return A;
141  }
