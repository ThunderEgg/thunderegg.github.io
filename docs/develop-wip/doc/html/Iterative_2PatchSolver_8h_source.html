210  auto f_copy = vg->getNewVector();

212  auto f_copy_lds = f_copy->getLocalDatas(0);
213  op->addGhostToRHS(pinfo, us, f_copy_lds);
214 
215  int iterations = 0;
216  try {
217  solver->solve(vg, single_op, u_single, f_copy);

221  }
222  }
