52  std::shared_ptr<Domain<D>> coarse_domain, int num_components,
53  bool extrapolate_boundary_ghosts = false)
54  : MPIRestrictor<D>(

59  void
60  restrictPatches(const std::vector<std::pair<int, std::shared_ptr<const PatchInfo<D>>>> &patches,
61  std::shared_ptr<const Vector<D>> finer_vector,
62  std::shared_ptr<Vector<D>> coarser_vector) const override
63  {
64  for (const auto &pair : patches) {
65  std::shared_ptr<const PatchInfo<D>> pinfo = pair.second;
66  auto coarse_local_datas = coarser_vector->getLocalDatas(pair.first);
67  auto fine_datas = finer_vector->getLocalDatas(pinfo->local_index);
68 
69  if (pinfo->hasCoarseParent()) {
70  Orthant<D> orth = pinfo->orth_on_parent;
71  // get starting index in coarser patch
72  std::array<int, D> starts;
73  for (size_t i = 0; i < D; i++) {
74  starts[i] = orth.isOnSide(Side<D>::LowerSideOnAxis(i))
75  ? 0
76  : coarse_local_datas[0].getLengths()[i];
77  }
78 
79  for (int c = 0; c < fine_datas.size(); c++) {
80  // interpolate interior values
81  nested_loop<D>(fine_datas[c].getStart(), fine_datas[c].getEnd(),
82  [&](const std::array<int, D> &coord) {
83  std::array<int, D> coarse_coord;
84  for (size_t x = 0; x < D; x++) {
85  coarse_coord[x] = (coord[x] + starts[x]) / 2;
86  }
87  coarse_local_datas[c][coarse_coord]
88  += fine_datas[c][coord] / (1 << D);
89  });
90 
91  if (extrapolate_boundary_ghosts) {
92  // extrapolate ghost values
93  for (Side<D> s : orth.getExteriorSides()) {
94  if (!pinfo->hasNbr(s)) {
95  auto fine_ghost = fine_datas[c].getGhostSliceOnSide(s, 1);
96  auto fine_interior = fine_datas[c].getSliceOnSide(s);
97  auto coarse_ghost = coarse_local_datas[c].getGhostSliceOnSide(s, 1);
98  nested_loop<D - 1>(
99  fine_ghost.getStart(), fine_ghost.getEnd(),
100  [&](const std::array<int, D - 1> &coord) {
101  std::array<int, D - 1> coarse_coord;
102  for (size_t x = 0; x < s.getAxisIndex(); x++) {
103  coarse_coord[x] = (coord[x] + starts[x]) / 2;
104  }
105  for (size_t x = s.getAxisIndex() + 1; x < D; x++) {
106  coarse_coord[x - 1] = (coord[x - 1] + starts[x]) / 2;
107  }
108  coarse_ghost[coarse_coord]
109  += (3 * fine_ghost[coord] - fine_interior[coord]) / (1 << D);
110  });
111  }
112  }
113  }
114  }
115  } else {
116  for (int c = 0; c < fine_datas.size(); c++) {
117  // just copy the values
118  nested_loop<D>(fine_datas[c].getStart(), fine_datas[c].getEnd(),
119  [&](const std::array<int, D> &coord) {
120  coarse_local_datas[c][coord] += fine_datas[c][coord];
121  });
122  if (extrapolate_boundary_ghosts) {
123  // copy boundary ghost values
124  for (Side<D> s : Side<D>::getValues()) {
125  if (!pinfo->hasNbr(s)) {
126  auto fine_ghost = fine_datas[c].getGhostSliceOnSide(s, 1);
127  auto coarse_ghost = coarse_local_datas[c].getGhostSliceOnSide(s, 1);
128  nested_loop<D - 1>(fine_ghost.getStart(), fine_ghost.getEnd(),
129  [&](const std::array<int, D - 1> &coord) {
130  coarse_ghost[coord] += fine_ghost[coord];
131  });
132  }
133  }
134  }
135  }
136  }
137  }
138  }
