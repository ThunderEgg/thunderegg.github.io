108 template <size_t D> class NormalIfaceInfo : public IfaceInfo<D>
109 {
110  public:
114  std::shared_ptr<NormalNbrInfo<D>> nbr_info;
118  NormalIfaceInfo()
119  {
120  this->id = 0;
121  this->local_index = 0;
122  this->global_index = 0;
123  }
130  NormalIfaceInfo(std::shared_ptr<PatchInfo<D>> pinfo, Side<D> s)
131  {
132  nbr_info = pinfo->getNormalNbrInfoPtr(s);
133  if (s.isLowerOnAxis()) {
134  this->id = pinfo->id * Side<D>::num_sides + s.getIndex();
135  // lower axis interface belongs to neighboring rank
136  this->rank = nbr_info->rank;
137  } else {
138  this->id = nbr_info->id * Side<D>::num_sides + s.opposite().getIndex();
139  // higher axis interafce belongs to this patch's rank
140  this->rank = pinfo->rank;
141  }
142  }
143  void getIds(std::deque<int> &ids)
144  {
145  ids.push_back(this->id);
146  }
147  void getLocalIndexes(std::deque<int> &idx)
148  {
149  idx.push_back(this->local_index);
150  }
151  void getGlobalIndexes(std::deque<int> &idx)
152  {
153  idx.push_back(this->global_index);
154  }
155  void getIfaceTypes(std::deque<IfaceType<D>> &types)
156  {
157  types.push_back(IfaceType<D>::normal);
158  }
159  void getRanks(std::deque<int> &ranks)
160  {
161  ranks.push_back(nbr_info->rank);
162  }
163  void setLocalIndexes(const std::map<int, int> &rev_map)
164  {
165  this->local_index = rev_map.at(this->id);
166  }
167  void setGlobalIndexes(const std::map<int, int> &rev_map)
168  {
169  this->global_index = rev_map.at(this->local_index);
170  }
171 };
180 template <size_t D> class CoarseIfaceInfo : public IfaceInfo<D>
181 {
182  public:
186  std::shared_ptr<CoarseNbrInfo<D>> nbr_info;
190  Orthant<D - 1> orth_on_coarse;
194  int coarse_rank;
198  int coarse_id;
202  int coarse_local_index;
206  int coarse_global_index;
213  CoarseIfaceInfo(std::shared_ptr<PatchInfo<D>> pinfo, Side<D> s)
214  {
215  nbr_info = pinfo->getCoarseNbrInfoPtr(s);
216  this->id = pinfo->id * Side<D>::num_sides + s.getIndex();
217  orth_on_coarse = nbr_info->orth_on_coarse;
218  coarse_id = nbr_info->id * Side<D>::num_sides + s.opposite().getIndex();
219  // fine and coarse interfaces always belong to their patches
220  this->rank = pinfo->rank;
221  this->coarse_rank = nbr_info->rank;
222  }
223  void getIds(std::deque<int> &ids)
224  {
225  ids.push_back(this->id);
226  ids.push_back(coarse_id);
227  }
228  void getLocalIndexes(std::deque<int> &idx)
229  {
230  idx.push_back(this->local_index);
231  idx.push_back(coarse_local_index);
232  }
233  void getGlobalIndexes(std::deque<int> &idx)
234  {
235  idx.push_back(this->global_index);
236  idx.push_back(coarse_global_index);
237  }
238  void getIfaceTypes(std::deque<IfaceType<D>> &types)
239  {
240  IfaceType<D> fine_type(IfaceType<D>::fine_to_fine, orth_on_coarse);
241  IfaceType<D> coarse_type(IfaceType<D>::fine_to_coarse, orth_on_coarse);
242  types.push_back(fine_type);
243  types.push_back(coarse_type);
244  }
245  void getRanks(std::deque<int> &ranks)
246  {
247  ranks.push_back(nbr_info->rank);
248  ranks.push_back(nbr_info->rank);
249  }
250  void setLocalIndexes(const std::map<int, int> &rev_map)
251  {
252  this->local_index = rev_map.at(this->id);
253  auto it = rev_map.find(coarse_id);
254  if (it != rev_map.end())
255  coarse_local_index = it->second;
256  }
257  void setGlobalIndexes(const std::map<int, int> &rev_map)
258  {
259  this->global_index = rev_map.at(this->local_index);
260  coarse_global_index = rev_map.at(coarse_local_index);
261  }

271 template <size_t D> class FineIfaceInfo : public IfaceInfo<D>
272 {
273  public:
277  std::shared_ptr<FineNbrInfo<D>> nbr_info;
281  std::array<int, Orthant<D - 1>::num_orthants> fine_ranks;
285  std::array<int, Orthant<D - 1>::num_orthants> fine_ids;
289  std::array<int, Orthant<D - 1>::num_orthants> fine_local_indexes;
293  std::array<int, Orthant<D - 1>::num_orthants> fine_global_indexes;
300  FineIfaceInfo(std::shared_ptr<PatchInfo<D>> pinfo, Side<D> s)
301  {
302  nbr_info = pinfo->getFineNbrInfoPtr(s);
303  this->id = pinfo->id * Side<D>::num_sides + s.getIndex();
304  this->rank = pinfo->rank;
305  for (size_t i = 0; i < fine_ids.size(); i++) {
306  fine_ids[i] = nbr_info->ids[i] * Side<D>::num_sides + s.opposite().getIndex();
307  fine_ranks[i] = nbr_info->ranks[i];
308  }
309  }
310  void getIdxAndTypes(std::deque<int> &idx, std::deque<IfaceType<D>> &types)
311  {
312  idx.push_back(this->local_index);
313  types.push_back(IfaceType<D>::coarse_to_coarse);
314  for (size_t i = 0; i < fine_local_indexes.size(); i++) {
315  idx.push_back(fine_local_indexes[i]);
316  IfaceType<D> type(IfaceType<D>::coarse_to_fine, i);
317  types.push_back(type);
318  }
319  }
320  void getIds(std::deque<int> &ids)
321  {
322  ids.push_back(this->id);
323  for (size_t i = 0; i < fine_ids.size(); i++) {
324  ids.push_back(fine_ids[i]);
325  }
326  }
327  void getLocalIndexes(std::deque<int> &idx)
328  {
329  idx.push_back(this->local_index);
330  for (size_t i = 0; i < fine_ids.size(); i++) {
331  idx.push_back(fine_local_indexes[i]);
332  }
333  }
334  void getGlobalIndexes(std::deque<int> &idx)
335  {
336  idx.push_back(this->global_index);
337  for (size_t i = 0; i < fine_ids.size(); i++) {
338  idx.push_back(fine_global_indexes[i]);
339  }
340  }
341  void getIfaceTypes(std::deque<IfaceType<D>> &types)
342  {
343  types.push_back(IfaceType<D>::coarse_to_coarse);
344  for (Orthant<D - 1> o : Orthant<D - 1>::getValues()) {
345  IfaceType<D> type(IfaceType<D>::coarse_to_fine, o);
346  types.push_back(type);
347  }
348  }
349  void getRanks(std::deque<int> &ranks)
350  {
351  ranks.push_back(-1);
352  for (size_t i = 0; i < fine_ids.size(); i++) {
353  ranks.push_back(nbr_info->ranks[i]);
354  }
355  }
356  void setLocalIndexes(const std::map<int, int> &rev_map)
357  {
358  this->local_index = rev_map.at(this->id);
359  for (size_t i = 0; i < fine_ids.size(); i++) {
360  auto it = rev_map.find(this->fine_ids[i]);
361  if (it != rev_map.end())
362  fine_local_indexes[i] = it->second;
363  }
364  }
365  void setGlobalIndexes(const std::map<int, int> &rev_map)
366  {
367  this->global_index = rev_map.at(this->local_index);
368  for (size_t i = 0; i < fine_local_indexes.size(); i++) {
369  fine_global_indexes[i] = rev_map.at(fine_local_indexes[i]);
370  }
371  }
372 };
378 template <size_t D> struct SchurInfo : public Serializable {
382  std::shared_ptr<PatchInfo<D>> pinfo;
386  std::array<std::shared_ptr<IfaceInfo<D>>, Side<D>::num_sides> iface_info;
391  SchurInfo()
392  {
393  iface_info.fill(nullptr);
394  }
400  SchurInfo(std::shared_ptr<PatchInfo<D>> &pinfo)
401  {
402  this->pinfo = pinfo;
403  iface_info.fill(nullptr);
404 
405  // create iface objects
406  for (Side<D> s : Side<D>::getValues()) {
407  if (pinfo->hasNbr(s)) {
408  switch (pinfo->getNbrType(s)) {
409  case NbrType::Normal:
410  getIfaceInfoPtr(s).reset(new NormalIfaceInfo<D>(pinfo, s));
411  break;
412  case NbrType::Fine:
413  getIfaceInfoPtr(s).reset(new FineIfaceInfo<D>(pinfo, s));
414  break;
415  case NbrType::Coarse:
416  getIfaceInfoPtr(s).reset(new CoarseIfaceInfo<D>(pinfo, s));
417  break;
418  }
419  }
420  }
421  }
427  std::shared_ptr<IfaceInfo<D>> &getIfaceInfoPtr(Side<D> s)
428  {
429  return iface_info[s.getIndex()];
430  }
431  /*
432  * @brief Get a reference to the NormalIfaceInfo object.
433  *
434  * If there is not a NormalIfaceInfo object on this side, the behavior is undefined.
435  *
436  * @param s the side of the patch that the interface is on
437  */
438  NormalIfaceInfo<D> &getNormalIfaceInfo(Side<D> s)
439  {
440  return *std::dynamic_pointer_cast<NormalIfaceInfo<D>>(iface_info[s.getIndex()]);
441  }
442  /*
443  * @brief Get a reference to the CoarseIfaceInfo object.
444  *
445  * If there is not a CoarseIfaceInfo object on this side, the behavior is undefined.
446  *
447  * @param s the side of the patch that the interface is on
448  */
449  CoarseIfaceInfo<D> &getCoarseIfaceInfo(Side<D> s)
450  {
451  return *std::dynamic_pointer_cast<CoarseIfaceInfo<D>>(iface_info[s.getIndex()]);
452  }
453  /*
454  * @brief Get a reference to the FineIfaceInfo object.
455  *
456  * If there is not a FineIfaceInfo object on this side, the behavior is undefined.
457  *
458  * @param s the side of the patch that the interface is on
459  */
460  FineIfaceInfo<D> &getFineIfaceInfo(Side<D> s)
461  {
462  return *std::dynamic_pointer_cast<FineIfaceInfo<D>>(iface_info[s.getIndex()]);
463  }
464  /*
465  * @brief Get a reference to the NormalIfaceInfo object.
466  *
467  * If there is not a NormalIfaceInfo object on this side, the behavior is undefined.
468  *
469  * @param s the side of the patch that the interface is on
470  */
471  const NormalIfaceInfo<D> &getNormalIfaceInfo(Side<D> s) const
472  {
473  return *std::dynamic_pointer_cast<NormalIfaceInfo<D>>(iface_info[s.getIndex()]);
474  }
475  /*
476  * @brief Get a reference to the CoarseIfaceInfo object.
477  *
478  * If there is not a CoarseIfaceInfo object on this side, the behavior is undefined.
479  *
480  * @param s the side of the patch that the interface is on
481  */
482  const CoarseIfaceInfo<D> &getCoarseIfaceInfo(Side<D> s) const
483  {
484  return *std::dynamic_pointer_cast<CoarseIfaceInfo<D>>(iface_info[s.getIndex()]);
485  }
486  /*
487  * @brief Get a reference to the FineIfaceInfo object.
488  *
489  * If there is not a FineIfaceInfo object on this side, the behavior is undefined.
490  *
491  * @param s the side of the patch that the interface is on
492  */
493  const FineIfaceInfo<D> &getFineIfaceInfo(Side<D> s) const
494  {
495  return *std::dynamic_pointer_cast<FineIfaceInfo<D>>(iface_info[s.getIndex()]);
496  }
505  std::deque<int> getIds()
506  {
507  std::deque<int> retval;
508  for (Side<D> s : Side<D>::getValues()) {
509  if (pinfo->hasNbr(s)) {
510  getIfaceInfoPtr(s)->getIds(retval);
511  }
512  }
513  return retval;
514  }
515  void setLocalIndexes(const std::map<int, int> &rev_map)
516  {
517  for (Side<D> s : Side<D>::getValues()) {
518  if (pinfo->hasNbr(s)) {
519  getIfaceInfoPtr(s)->setLocalIndexes(rev_map);
520  }
521  }
522  }
523  void setGlobalIndexes(const std::map<int, int> &rev_map)
524  {
525  for (Side<D> s : Side<D>::getValues()) {
526  if (pinfo->hasNbr(s)) {
527  getIfaceInfoPtr(s)->setGlobalIndexes(rev_map);
528  }
529  }
530  }
531  int getIfaceLocalIndex(Side<D> s) const
532  {
533  return iface_info[s.getIndex()]->local_index;
534  }
535  int serialize(char *buffer) const
536  {
537  BufferWriter writer(buffer);
538  writer << *pinfo;
539  return writer.getPos();
540  }
541  int deserialize(char *buffer)
542  {
543  BufferReader reader(buffer);
544  pinfo.reset(new PatchInfo<D>());
545  reader >> *pinfo;
546  // create iface objects
547  for (Side<D> s : Side<D>::getValues()) {
548  if (pinfo->hasNbr(s)) {
549  switch (pinfo->getNbrType(s)) {
550  case NbrType::Normal:
551  getIfaceInfoPtr(s).reset(new NormalIfaceInfo<D>(pinfo, s));
552  break;
553  case NbrType::Fine:
554  getIfaceInfoPtr(s).reset(new FineIfaceInfo<D>(pinfo, s));
555  break;
556  case NbrType::Coarse:
557  getIfaceInfoPtr(s).reset(new CoarseIfaceInfo<D>(pinfo, s));
558  break;
559  }
560  }
561  }
562  return reader.getPos();
563  }
