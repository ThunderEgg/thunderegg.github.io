113  void setIncomingBufferMapsAndDetermineLocalVectorSize(
114  std::shared_ptr<const InterfaceDomain<D>> iface_domain)
115  {
116  int rank;
117  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
118 
119  std::map<int, std::set<std::pair<int, int>>> incoming_ranks_to_id_local_index_pairs;
120 
121  for (auto piinfo : iface_domain->getPatchIfaceInfos()) {
122  for (Side<D> s : Side<D>::getValues()) {
123  if (piinfo->pinfo->hasNbr(s)) {
124  auto iface_info = piinfo->getIfaceInfo(s);
125  if (iface_info->rank != rank) {
126  incoming_ranks_to_id_local_index_pairs[iface_info->rank].emplace(
127  iface_info->id, iface_info->patch_local_index);
128  }
129  }
130  }
131  }
132  int local_vector_size = 0;
133  for (auto rank_to_id_local_index_pairs : incoming_ranks_to_id_local_index_pairs) {
134  local_vector_size += rank_to_id_local_index_pairs.second.size();
135  }
136  local_vector_size += iface_domain->getNumLocalInterfaces();
137  num_local_patch_ifaces = local_vector_size;
138 
139  num_recvs = incoming_ranks_to_id_local_index_pairs.size();
140  recv_ranks.resize(num_recvs);
141  recv_requests.resize(num_recvs);
142  recv_local_indexes.resize(num_recvs);
143  recv_buffers.resize(num_recvs);
144 
145  int recv_index = 0;
146  for (const auto &rank_to_id_local_index_pairs : incoming_ranks_to_id_local_index_pairs) {
147  recv_ranks[recv_index] = rank_to_id_local_index_pairs.first;
148  std::vector<int> &local_index_vector = recv_local_indexes[recv_index];
149 
150  local_index_vector.reserve(rank_to_id_local_index_pairs.second.size());
151 
152  for (auto id_local_index_pair : rank_to_id_local_index_pairs.second) {
153  local_index_vector.push_back(id_local_index_pair.second);
154  }
155  recv_index++;
156  }
157  }
163  void setOutgoingBufferMaps(std::shared_ptr<const InterfaceDomain<D>> iface_domain)
164  {
165  int rank;
166  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
167 
168  std::map<int, std::set<std::pair<int, int>>> outgoing_ranks_to_id_local_index_pairs;
169  for (auto iface : iface_domain->getInterfaces()) {
170  for (auto patch : iface->patches) {
171  if ((patch.type.isNormal() || patch.type.isFineToFine()
172  || patch.type.isCoarseToCoarse())
173  && patch.piinfo->pinfo->rank != rank) {
174  outgoing_ranks_to_id_local_index_pairs[patch.piinfo->pinfo->rank].emplace(
175  iface->id, iface->local_index);
176  }
177  }
178  }
179 
180  num_sends = outgoing_ranks_to_id_local_index_pairs.size();
181  send_ranks.resize(num_sends);
182  send_requests.resize(num_sends);
183  send_local_indexes.resize(num_sends);
184  send_buffers.resize(num_sends);
185 
186  int send_index = 0;
187  for (const auto &rank_to_id_local_index_pairs : outgoing_ranks_to_id_local_index_pairs) {
188  send_ranks[send_index] = rank_to_id_local_index_pairs.first;
189  std::vector<int> &local_index_vector = send_local_indexes[send_index];
190 
191  local_index_vector.reserve(rank_to_id_local_index_pairs.second.size());
192 
193  for (auto id_local_index_pair : rank_to_id_local_index_pairs.second) {
194  local_index_vector.push_back(id_local_index_pair.second);
195  }
196  send_index++;
197  }
198  }
202  void initializeMPIBuffers()
203  {
204  for (int send_index = 0; send_index < num_sends; send_index++) {
205  send_buffers[send_index].resize(send_local_indexes[send_index].size() * iface_stride);

209  recv_buffers[recv_index].resize(recv_local_indexes[recv_index].size() * iface_stride);
210  }

218  send_buffers[send_index] = std::vector<double>();
219  }

231  explicit PatchIfaceScatter(std::shared_ptr<const InterfaceDomain<D>> iface_domain)
232  {
233  std::array<int, D> ns = iface_domain->getDomain()->getNs();
234  for (int i = 1; i < D; i++) {
235  if (ns[0] != ns[i]) {
236  throw RuntimeError(
237  "Cannot form Schur compliment vector for Domain with non-square patches");
238  }
239  }
240 
241  lengths.fill(ns[0]);
242  iface_stride = std::pow(ns[0], D - 1);
243  setIncomingBufferMapsAndDetermineLocalVectorSize(iface_domain);
244  setOutgoingBufferMaps(iface_domain);
245  }

278  void scatterStart(std::shared_ptr<const Vector<D - 1>> global_vector,
279  std::shared_ptr<Vector<D - 1>> local_patch_iface_vector)
280  {
281  if (communicating) {
282  throw RuntimeError("This PatchIfaceScatter is in the middle of communicating");
283  }
284 
285  for (int i = 0; i < global_vector->getNumLocalPatches(); i++) {
286  auto global_data = global_vector->getLocalData(i);
287  auto local_data = local_patch_iface_vector->getLocalData(i);
288  nested_loop<D - 1>(
289  local_data.getStart(), local_data.getEnd(),
290  [&](const std::array<int, D - 1> &coord) { local_data[coord] = global_data[coord]; });
291  }
292 
293  initializeMPIBuffers();
294 
295  for (int recv_index = 0; recv_index < num_recvs; recv_index++) {
296  MPI_Irecv(recv_buffers[recv_index].data(), recv_buffers[recv_index].size(), MPI_DOUBLE,
297  recv_ranks[recv_index], 0, MPI_COMM_WORLD, &recv_requests[recv_index]);
298  }
299 
300  for (int send_index = 0; send_index < num_sends; send_index++) {
301  std::vector<double> &buffer = send_buffers[send_index];
302 
303  int buffer_index = 0;
304  for (int local_index : send_local_indexes[send_index]) {
305  auto local_data = global_vector->getLocalData(local_index);
306  nested_loop<D - 1>(local_data.getStart(), local_data.getEnd(),
307  [&](const std::array<int, D - 1> &coord) {
308  buffer[buffer_index] = local_data[coord];
309  buffer_index++;
310  });
311  }
312 
313  MPI_Isend(buffer.data(), buffer.size(), MPI_DOUBLE, send_ranks[send_index], 0,
314  MPI_COMM_WORLD, &send_requests[send_index]);
315  }
316 
317  for (int local_iface = 0; local_iface < global_vector->getNumLocalPatches();
318  local_iface++) {
319  auto global_data = global_vector->getLocalData(local_iface);
320  auto local_data = local_patch_iface_vector->getLocalData(local_iface);
321  nested_loop<D - 1>(
322  local_data.getStart(), local_data.getEnd(),
323  [&](const std::array<int, D - 1> &coord) { local_data[coord] = global_data[coord]; });
324  }
325 
326  curr_global_vector = global_vector;
327  curr_local_vector = local_patch_iface_vector;
328  communicating = true;
329  }
340  void scatterFinish(std::shared_ptr<const Vector<D - 1>> global_vector,
341  std::shared_ptr<Vector<D - 1>> local_patch_iface_vector)
342  {
343  if (global_vector != curr_global_vector || local_patch_iface_vector != curr_local_vector) {
344  throw RuntimeError(
345  "Different vectors were passed ot scatterFinish than were passed to scatterStart");
346  }
347 
348  for (int i = 0; i < num_recvs; i++) {
349  MPI_Status status;
350  int recv_index;
351  MPI_Waitany(num_recvs, recv_requests.data(), &recv_index, &status);
352 
353  std::vector<double> &buffer = recv_buffers[recv_index];
354 
355  int buffer_index = 0;
356  for (int local_index : recv_local_indexes[recv_index]) {
357  auto local_data = local_patch_iface_vector->getLocalData(local_index);
358  nested_loop<D - 1>(local_data.getStart(), local_data.getEnd(),
359  [&](const std::array<int, D - 1> &coord) {
360  local_data[coord] = buffer[buffer_index];
361  buffer_index++;
362  });
363  }
364  }
365 
366  MPI_Waitall(num_sends, send_requests.data(), MPI_STATUSES_IGNORE);
367 
368  curr_global_vector = nullptr;
369  curr_local_vector = nullptr;
370  communicating = false;
371  destroyMPIBuffers();
372  }
