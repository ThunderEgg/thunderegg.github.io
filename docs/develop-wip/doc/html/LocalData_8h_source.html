86  LocalData<D - 1> getSliceOnSidePriv(Side<D> s, int offset) const
87  {
88  size_t axis = s.getAxisIndex();
89  std::array<int, D - 1> new_strides;
90  for (size_t i = 0; i < axis; i++) {
91  new_strides[i] = strides[i];
92  }
93  for (size_t i = axis; i < D - 1; i++) {
94  new_strides[i] = strides[i + 1];
95  }
96  std::array<int, D - 1> new_lengths;
97  for (size_t i = 0; i < axis; i++) {
98  new_lengths[i] = lengths[i];
99  }
100  for (size_t i = axis; i < D - 1; i++) {
101  new_lengths[i] = lengths[i + 1];
102  }
103  if (s.isLowerOnAxis()) {
104  double *new_data = data + offset * strides[axis];
105  return LocalData<D - 1>(new_data, new_strides, new_lengths, 0, ldm);
106  } else {
107  double *new_data = data + (lengths[axis] - 1 - offset) * strides[axis];
108  return LocalData<D - 1>(new_data, new_strides, new_lengths, 0, ldm);
109  }
110  }
