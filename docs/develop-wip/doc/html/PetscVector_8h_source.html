87  public:
88  PetscVector(Vec vec, int num_local_patches, const std::array<int, D> &lengths,
89  int num_ghost_cells, bool own = true)
90  : Vector<D>(MPI_COMM_WORLD)
91  {
92  this->own = own;
93  this->lengths = lengths;
94  this->vec = vec;
95  this->num_ghost_cells = num_ghost_cells;
96 
97  strides[0] = 1;
98  first_offset = num_ghost_cells;
99  for (size_t i = 1; i < D; i++) {
100  strides[i] = (this->lengths[i - 1] + 2 * num_ghost_cells) * strides[i - 1];
101  first_offset += strides[i] * num_ghost_cells;
102  }
103  patch_stride = strides[D - 1] * (lengths[D - 1] + 2 * num_ghost_cells);
104  if (num_local_patches == -1) {
105  VecGetLocalSize(vec, &this->num_local_patches);
106  this->num_local_patches /= patch_stride;
107  } else {
108  this->num_local_patches = num_local_patches;
109  }
110  int num_cells_in_patch = 1;
111  for (size_t i = 0; i < D; i++) {
112  num_cells_in_patch *= this->lengths[i];
113  }
114  this->num_local_cells = this->num_local_patches * num_cells_in_patch;
115  }
