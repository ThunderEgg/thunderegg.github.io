60  void prepCoarser(const Level<D> &level, VecList &u_vectors, ConstVecList &f_vectors) const
61  {
62  // calculate residual
63  std::shared_ptr<Vector<D>> r = level.getVectorGenerator()->getNewVector();
64  level.getOperator().apply(u_vectors.front(), r);
65  r->scaleThenAdd(-1, f_vectors.front());
66  // create vectors for coarser levels
67  std::shared_ptr<Vector<D>> new_u = level.getCoarser()->getVectorGenerator()->getNewVector();
68  std::shared_ptr<Vector<D>> new_f = level.getCoarser()->getVectorGenerator()->getNewVector();
69  level.getRestrictor().restrict(new_f, r);
70  u_vectors.push_front(new_u);

72  }

84  }
