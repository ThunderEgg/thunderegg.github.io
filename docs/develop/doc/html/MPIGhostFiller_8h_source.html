88  LocalData<D> getLocalDataForBuffer(double *buffer_ptr, const Side<D> side,
89  int component_index) const
90  {
91  auto ns = domain->getNs();
92  int num_ghost_cells = domain->getNumGhostCells();
93  // determine striding
94  std::array<int, D> strides;
95  strides[0] = 1;
96  for (size_t i = 1; i < D; i++) {
97  if (i - 1 == side.getAxisIndex()) {
98  strides[i] = num_ghost_cells * strides[i - 1];
99  } else {
100  strides[i] = ns[i - 1] * strides[i - 1];
101  }
102  }
103  int size = D - 1 == side.getAxisIndex() ? (num_ghost_cells * strides[D - 1])
104  : (ns[D - 1] * strides[D - 1]);
105  // transform buffer ptr so that it points to first non-ghost cell
106  double *transformed_buffer_ptr = buffer_ptr + size * component_index;
107  if (side.isLowerOnAxis()) {
108  transformed_buffer_ptr -= (-num_ghost_cells) * strides[side.getAxisIndex()];
109  } else {
110  transformed_buffer_ptr -= ns[side.getAxisIndex()] * strides[side.getAxisIndex()];
111  }
112 
113  LocalData<D> buffer_data(transformed_buffer_ptr, strides, ns, num_ghost_cells);
114  return buffer_data;
115  }
122  std::vector<MPI_Request> postRecvs(std::vector<std::vector<double>> &buffers) const
123  {
124  std::vector<MPI_Request> recv_requests(buffers.size());
125  for (size_t i = 0; i < recv_requests.size(); i++) {
126  MPI_Irecv(buffers[i].data(), buffers[i].size(), MPI_DOUBLE, index_rank_map[i], 0,
127  MPI_COMM_WORLD, &recv_requests[i]);
128  }

138  void processRecvs(std::vector<MPI_Request> &requests, std::vector<std::vector<double>> &buffers,
139  std::shared_ptr<const Vector<D>> u) const
140  {
141  size_t num_requests = requests.size();
142  for (size_t i = 0; i < num_requests; i++) {
143  int finished_index;
144  MPI_Waitany(requests.size(), requests.data(), &finished_index, MPI_STATUS_IGNORE);
145  for (auto t : incoming_ghosts[finished_index]) {
146  int local_index = std::get<0>(t);
147  Side<D> side = std::get<1>(t);
148  size_t buffer_offset = std::get<2>(t);
149 
150  for (int c = 0; c < u->getNumComponents(); c++) {
151  const LocalData<D> local_data = u->getLocalData(c, local_index);
152  double * buffer_ptr
153  = buffers[finished_index].data() + buffer_offset * u->getNumComponents();
154  LocalData<D> buffer_data = getLocalDataForBuffer(buffer_ptr, side, c);
155  for (int ig = 0; ig < domain->getNumGhostCells(); ig++) {
156  LocalData<D - 1> local_slice = local_data.getGhostSliceOnSide(side, ig + 1);
157  LocalData<D - 1> buffer_slice
158  = buffer_data.getGhostSliceOnSide(side, ig + 1);
159  nested_loop<D - 1>(local_slice.getStart(), local_slice.getEnd(),
160  [&](const std::array<int, D - 1> &coord) {
161  local_slice[coord] += buffer_slice[coord];
162  });
163  }
164  }
165  }
166  }
167  }
175  std::vector<MPI_Request> postSends(std::vector<std::vector<double>> &buffers,
176  std::shared_ptr<const Vector<D>> u) const
177  {
178  std::vector<MPI_Request> send_requests(send_buff_lengths.size());
179  for (size_t i = 0; i < remote_calls.size(); i++) {
180  for (const RemoteCall &call : remote_calls[i]) {
181  auto pinfo = std::get<0>(call);
182  auto side = std::get<1>(call);
183  auto nbr_type = std::get<2>(call);
184  auto orthant = std::get<3>(call);
185  auto local_datas = u->getLocalDatas(std::get<4>(call));
186  size_t buffer_offset = std::get<5>(call);
187  double *buffer_ptr = buffers[i].data() + buffer_offset * u->getNumComponents();
188 
189  // create LocalData objects for the buffer
190  std::vector<LocalData<D>> buffer_datas(u->getNumComponents());
191  for (int c = 0; c < u->getNumComponents(); c++) {
192  buffer_datas[c] = getLocalDataForBuffer(buffer_ptr, side.opposite(), c);
193  }
194 
195  // make the call
196  fillGhostCellsForNbrPatch(pinfo, local_datas, buffer_datas, side, nbr_type,
197  orthant);
198  }
199  MPI_Isend(buffers[i].data(), buffers[i].size(), MPI_DOUBLE, index_rank_map[i], 0,
200  MPI_COMM_WORLD, &send_requests[i]);
201  }
202  return send_requests;
203  }

222  MPIGhostFiller(std::shared_ptr<const Domain<D>> domain_in, int side_cases_in)
223  : domain(domain_in), side_cases(side_cases_in)
224  {
225  int rank;
226  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
227  // rank nbr_id side orthant nbrtype pinfo local_index
228  std::set<std::tuple<int, int, const Side<D>, const Orthant<D>, const NbrType,
229  std::shared_ptr<const PatchInfo<D>>, int>>
230  remote_call_set;
231  // rank id side local index
232  std::set<std::tuple<int, int, const Side<D>, int>> incoming_ghost_set;
233 
234  std::array<size_t, D> axis_ghost_lengths;
235  for (size_t axis = 0; axis < D; axis++) {
236  size_t length = 1;
237  for (size_t i = 0; i < D; i++) {
238  if (i == axis) {
239  length *= domain->getNumGhostCells();
240  } else {
241  length *= domain->getNs()[i];
242  }
243  }
244  axis_ghost_lengths[axis] = length;
245  }
246  std::set<int> ranks;
247  for (auto pinfo : domain->getPatchInfoVector()) {
248  for (Side<D> s : Side<D>::getValues()) {
249  if (pinfo->hasNbr(s)) {
250  switch (pinfo->getNbrType(s)) {
251  case NbrType::Normal: {
252  auto nbrinfo = pinfo->getNormalNbrInfo(s);
253  if (nbrinfo.rank == rank) {
254  local_calls.emplace_back(pinfo, s, NbrType::Normal,
255  Orthant<D>::null(), pinfo->local_index,
256  nbrinfo.local_index);
257  } else {
258  ranks.insert(nbrinfo.rank);
259  remote_call_set.emplace(nbrinfo.rank, nbrinfo.id, s.opposite(),
260  Orthant<D>::null(), NbrType::Normal, pinfo,
261  pinfo->local_index);
262  incoming_ghost_set.emplace(nbrinfo.rank, pinfo->id, s,
263  pinfo->local_index);
264  }
265  } break;
266  case NbrType::Fine: {
267  auto nbrinfo = pinfo->getFineNbrInfo(s);
268  auto orthants = Orthant<D>::getValuesOnSide(s);
269  for (size_t i = 0; i < orthants.size(); i++) {
270  if (nbrinfo.ranks[i] == rank) {
271  local_calls.emplace_back(pinfo, s, NbrType::Fine, orthants[i],
272  pinfo->local_index,
273  nbrinfo.local_indexes[i]);
274  } else {
275  ranks.insert(nbrinfo.ranks[i]);
276  remote_call_set.emplace(
277  nbrinfo.ranks[i], nbrinfo.ids[i], s.opposite(), orthants[i],
278  NbrType::Fine, pinfo, pinfo->local_index);
279  incoming_ghost_set.emplace(nbrinfo.ranks[i], pinfo->id, s,
280  pinfo->local_index);
281  }
282  }
283  } break;
284  case NbrType::Coarse: {
285  auto nbrinfo = pinfo->getCoarseNbrInfo(s);
286  auto orthant = Orthant<D>::getValuesOnSide(
287  s.opposite())[nbrinfo.orth_on_coarse.getIndex()];
288  if (nbrinfo.rank == rank) {
289  local_calls.emplace_back(pinfo, s, NbrType::Coarse, orthant,
290  pinfo->local_index, nbrinfo.local_index);
291  } else {
292  ranks.insert(nbrinfo.rank);
293  remote_call_set.emplace(nbrinfo.rank, nbrinfo.id, s.opposite(),
294  orthant, NbrType::Coarse, pinfo,
295  pinfo->local_index);
296  incoming_ghost_set.emplace(nbrinfo.rank, pinfo->id, s,
297  pinfo->local_index);
298  }
299  } break;
300  }
301  }
302  }
303  }
304  std::map<int, size_t> rank_index_map;
305  index_rank_map.reserve(ranks.size());
306  int curr_index = 0;
307  for (int rank : ranks) {
308  rank_index_map[rank] = curr_index;
309  index_rank_map.push_back(rank);
310  curr_index++;
311  }
312  std::tuple<int, Side<D>> prev_id_side;
313  send_buff_lengths.resize(ranks.size());
314  remote_calls.resize(ranks.size());
315  for (auto call : remote_call_set) {
316  int local_buffer_index = rank_index_map[std::get<0>(call)];
317  int id = std::get<1>(call);
318  auto side = std::get<2>(call).opposite();
319  auto orthant = std::get<3>(call);
320  auto nbr_type = std::get<4>(call);
321  auto pinfo = std::get<5>(call);
322  auto local_index = std::get<6>(call);
323 
324  size_t offset = send_buff_lengths[local_buffer_index];
325 
326  // calculate length in buffer need for ghost cells
327  size_t length = axis_ghost_lengths[side.getAxisIndex()];
328  // add length to buffer length
329  // if its the same side of the patch, resuse the previous buffer space
330  if (std::make_tuple(id, side) == prev_id_side) {
331  offset -= length;
332  } else {
333  send_buff_lengths[local_buffer_index] += length;
334  }
335 
336  remote_calls[local_buffer_index].emplace_back(pinfo, side, nbr_type, orthant,
337  local_index, offset);
338  prev_id_side = std::make_tuple(id, side);
339  }
340  recv_buff_lengths.resize(ranks.size());
341  incoming_ghosts.resize(ranks.size());
342  for (auto t : incoming_ghost_set) {
343  int local_buffer_index = rank_index_map[std::get<0>(t)];
344  Side<D> side = std::get<2>(t);
345  int local_index = std::get<3>(t);
346 
347  // add length for ghosts to buffer length
348  size_t length = axis_ghost_lengths[side.getAxisIndex()];
349  size_t offset = recv_buff_lengths[local_buffer_index];
350  recv_buff_lengths[local_buffer_index] += length;
351 
352  // add ghost to incoming ghosts
353  incoming_ghosts[local_buffer_index].emplace_back(local_index, side, offset);
354  }
355  }

391  void fillGhost(std::shared_ptr<const Vector<D>> u) const
392  {
393  // zero out ghost cells
394  for (auto pinfo : domain->getPatchInfoVector()) {
395  for (auto &this_patch : u->getLocalDatas(pinfo->local_index)) {
396  for (Side<D> s : Side<D>::getValues()) {
397  if (pinfo->hasNbr(s)) {
398  for (int i = 0; i < pinfo->num_ghost_cells; i++) {
399  auto this_ghost = this_patch.getGhostSliceOnSide(s, i + 1);
400  nested_loop<D - 1>(
401  this_ghost.getStart(), this_ghost.getEnd(),
402  [&](const std::array<int, D - 1> &coord) { this_ghost[coord] = 0; });
403  }
404  }
405  }
406  }
407  }
408 
409  // allocate recv buffers and post recvs
410  std::vector<std::vector<double>> recv_buffers(recv_buff_lengths.size());
411  for (size_t i = 0; i < recv_buff_lengths.size(); i++) {
412  recv_buffers[i].resize(recv_buff_lengths[i] * u->getNumComponents());
413  }
414  std::vector<MPI_Request> recv_requests = postRecvs(recv_buffers);
415 
416  // allocate send buffers
417  std::vector<std::vector<double>> out_buffers(send_buff_lengths.size());
418  for (size_t i = 0; i < send_buff_lengths.size(); i++) {
419  out_buffers[i].resize(send_buff_lengths[i] * u->getNumComponents());
420  }
421  std::vector<MPI_Request> send_requests = postSends(out_buffers, u);
422 
423  // perform local operations
424  for (auto pinfo : domain->getPatchInfoVector()) {
425  auto datas = u->getLocalDatas(pinfo->local_index);
426  fillGhostCellsForLocalPatch(pinfo, datas);
427  }
428  for (const LocalCall &call : local_calls) {
429  auto pinfo = std::get<0>(call);
430  auto side = std::get<1>(call);
431  auto nbr_type = std::get<2>(call);
432  auto orthant = std::get<3>(call);
433  auto local_datas = u->getLocalDatas(std::get<4>(call));
434  auto nbr_datas = u->getLocalDatas(std::get<5>(call));
435  fillGhostCellsForNbrPatch(pinfo, local_datas, nbr_datas, side, nbr_type, orthant);
436  }
437 
438  processRecvs(recv_requests, recv_buffers, u);
439 
440  // wait for sends for finish
441  MPI_Waitall(send_requests.size(), send_requests.data(), MPI_STATUS_IGNORE);
442  }
