46  int max_iterations = 1000;
50  double tolerance = 1e-12;

56  void applyWithPreconditioner(std::shared_ptr<VectorGenerator<D>> vg,
57  std::shared_ptr<const Operator<D>> M_l,
58  std::shared_ptr<const Operator<D>> A,
59  std::shared_ptr<const Operator<D>> M_r,

130  int solve(std::shared_ptr<VectorGenerator<D>> vg, std::shared_ptr<const Operator<D>> A,
131  std::shared_ptr<Vector<D>> x, std::shared_ptr<const Vector<D>> b,
132  std::shared_ptr<const Operator<D>> Mr = nullptr, bool output = false,
133  std::ostream &os = std::cout) const override
134  {
135  std::shared_ptr<Vector<D>> resid = vg->getNewVector();
136 
137  A->apply(x, resid);
138  resid->scaleThenAdd(-1, b);
139 
140  std::shared_ptr<Vector<D>> initial_guess = vg->getNewVector();
141  initial_guess->copy(x);
142  x->set(0);
143 
144  double r0_norm = b->twoNorm();
145  std::shared_ptr<Vector<D>> rhat = vg->getNewVector();
146  rhat->copy(resid);
147  std::shared_ptr<Vector<D>> p = vg->getNewVector();
148  p->copy(resid);
149  std::shared_ptr<Vector<D>> ap = vg->getNewVector();
150  std::shared_ptr<Vector<D>> as = vg->getNewVector();
151 
152  std::shared_ptr<Vector<D>> s = vg->getNewVector();
153  double rho = rhat->dot(resid);
154 
155  int num_its = 0;
156  if (r0_norm == 0) {
157  return num_its;
158  }
159  double residual = resid->twoNorm() / r0_norm;
160  if (output) {
161  char buf[100];
162  sprintf(buf, "%5d %16.8e\n", num_its, residual);
163  os << std::string(buf);
164  }
165  while (residual > tolerance && num_its < max_iterations) {
166  if (timer) {
167  timer->start("Iteration");
168  }
169 
170  if (rho == 0) {
171  throw BreakdownError("BiCGStab broke down, rho was 0 on iteration "
172  + std::to_string(num_its));
173  }
174 
175  applyWithPreconditioner(vg, nullptr, A, Mr, p, ap);
176  double alpha = rho / rhat->dot(ap);
177  s->copy(resid);
178  s->addScaled(-alpha, ap);
179  if (s->twoNorm() / r0_norm <= tolerance) {
180  x->addScaled(alpha, p);
181  if (timer) {
182  timer->stop("Iteration");
183  }
184  break;
185  }
186  applyWithPreconditioner(vg, nullptr, A, Mr, s, as);
187  double omega = as->dot(s) / as->dot(as);
188  x->addScaled(alpha, p, omega, s);
189  resid->addScaled(-alpha, ap);
190  resid->addScaled(-omega, as);
191 
192  double rho_new = resid->dot(rhat);
193  double beta = rho_new * alpha / (rho * omega);
194  p->addScaled(-omega, ap);
195  p->scaleThenAdd(beta, resid);
196 
197  num_its++;
198  rho = rho_new;
199  residual = resid->twoNorm() / r0_norm;
200 
201  if (residual > 1e6) {
202  throw DivergenceError("BiCGStab reached divergence criteria on iteration "
203  + std::to_string(num_its) + " with residual two norm "
204  + std::to_string(residual));
205  }
206  if (output) {
207  char buf[100];
208  sprintf(buf, "%5d %16.8e\n", num_its, residual);
209  os << std::string(buf);
210  }
211  if (timer) {
212  timer->stop("Iteration");
213  }
214  }
215  if (Mr != nullptr) {
216  Mr->apply(x, resid);
217  x->copy(resid);
218  }
219  x->add(initial_guess);
220  return num_its;
221  }
