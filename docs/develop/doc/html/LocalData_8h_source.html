87  LocalData<D - 1> getSliceOnSidePriv(Side<D> s, int offset) const
88  {
89  size_t axis = s.getAxisIndex();
90  std::array<int, D - 1> new_strides;
91  for (size_t i = 0; i < axis; i++) {
92  new_strides[i] = strides[i];
93  }
94  for (size_t i = axis; i < D - 1; i++) {
95  new_strides[i] = strides[i + 1];
96  }
97  std::array<int, D - 1> new_lengths;
98  for (size_t i = 0; i < axis; i++) {
99  new_lengths[i] = lengths[i];
100  }
101  for (size_t i = axis; i < D - 1; i++) {
102  new_lengths[i] = lengths[i + 1];
103  }
104  if (s.isLowerOnAxis()) {
105  double *new_data = data + offset * strides[axis];
106  return LocalData<D - 1>(new_data, new_strides, new_lengths, 0, ldm);
107  } else {
108  double *new_data = data + (lengths[axis] - 1 - offset) * strides[axis];
109  return LocalData<D - 1>(new_data, new_strides, new_lengths, 0, ldm);
110  }
111  }
