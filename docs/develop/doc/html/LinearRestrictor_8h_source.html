48  void extrapolateBoundaries(std::shared_ptr<const PatchInfo<D>> pinfo,
49  const LocalData<D> &fine_data, LocalData<D> &coarse_data) const
50  {
51  Orthant<D> orth = pinfo->orth_on_parent;
52  std::array<int, D> starts;
53  for (size_t i = 0; i < D; i++) {
54  starts[i]
55  = orth.isOnSide(Side<D>::LowerSideOnAxis(i)) ? 0 : coarse_data.getLengths()[i];
56  }
57  // extrapolate ghost values
58  for (Side<D> s : pinfo->orth_on_parent.getExteriorSides()) {
59  if (!pinfo->hasNbr(s)) {
60  auto fine_ghost = fine_data.getGhostSliceOnSide(s, 1);
61  auto fine_interior = fine_data.getSliceOnSide(s);
62  auto coarse_ghost = coarse_data.getGhostSliceOnSide(s, 1);
63  nested_loop<D - 1>(fine_ghost.getStart(), fine_ghost.getEnd(),
64  [&](const std::array<int, D - 1> &coord) {
65  std::array<int, D - 1> coarse_coord;
66  for (size_t x = 0; x < s.getAxisIndex(); x++) {
67  coarse_coord[x] = (coord[x] + starts[x]) / 2;
68  }
69  for (size_t x = s.getAxisIndex() + 1; x < D; x++) {
70  coarse_coord[x - 1] = (coord[x - 1] + starts[x]) / 2;
71  }
72  coarse_ghost[coarse_coord]
73  += (3 * fine_ghost[coord] - fine_interior[coord]) / (1 << D);
74  });
75  }
76  }
77  }
86  void restrictToCoarserParent(std::shared_ptr<const PatchInfo<D>> pinfo, int parent_index,
87  std::shared_ptr<const Vector<D>> finer_vector,
88  std::shared_ptr<Vector<D>> coarser_vector) const
89  {
90  auto coarse_local_datas = coarser_vector->getLocalDatas(parent_index);
91  auto fine_datas = finer_vector->getLocalDatas(pinfo->local_index);
92  // get starting index in coarser patch
93  Orthant<D> orth = pinfo->orth_on_parent;
94  std::array<int, D> starts;
95  for (size_t i = 0; i < D; i++) {
96  starts[i] = orth.isOnSide(Side<D>::LowerSideOnAxis(i))
97  ? 0
98  : coarse_local_datas[0].getLengths()[i];
99  }
100 
101  for (size_t c = 0; c < fine_datas.size(); c++) {
102  // interpolate interior values
103  nested_loop<D>(
104  fine_datas[c].getStart(), fine_datas[c].getEnd(), [&](const std::array<int, D> &coord) {
105  std::array<int, D> coarse_coord;
106  for (size_t x = 0; x < D; x++) {
107  coarse_coord[x] = (coord[x] + starts[x]) / 2;
108  }
109  coarse_local_datas[c][coarse_coord] += fine_datas[c][coord] / (1 << D);
110  });
111 
112  if (extrapolate_boundary_ghosts) {
113  extrapolateBoundaries(pinfo, fine_datas[c], coarse_local_datas[c]);
114  }
115  }
116  }
126  void copyToParent(std::shared_ptr<const PatchInfo<D>> pinfo, int parent_index,
127 
128  std::shared_ptr<const Vector<D>> finer_vector,
129  std::shared_ptr<Vector<D>> coarser_vector) const
130  {
131  auto coarse_local_datas = coarser_vector->getLocalDatas(parent_index);
132  auto fine_datas = finer_vector->getLocalDatas(pinfo->local_index);
133  for (size_t c = 0; c < fine_datas.size(); c++) {
134  // just copy the values
135  nested_loop<D>(fine_datas[c].getStart(), fine_datas[c].getEnd(),
136  [&](const std::array<int, D> &coord) {
137  coarse_local_datas[c][coord] += fine_datas[c][coord];
138  });
139  if (extrapolate_boundary_ghosts) {
140  // copy boundary ghost values
141  for (Side<D> s : Side<D>::getValues()) {
142  if (!pinfo->hasNbr(s)) {
143  auto fine_ghost = fine_datas[c].getGhostSliceOnSide(s, 1);
144  auto coarse_ghost = coarse_local_datas[c].getGhostSliceOnSide(s, 1);
145  nested_loop<D - 1>(fine_ghost.getStart(), fine_ghost.getEnd(),
146  [&](const std::array<int, D - 1> &coord) {
147  coarse_ghost[coord] += fine_ghost[coord];
148  });
149  }
150  }
151  }
152  }
153  }
