36 template <int D> class AvgRstr : public MPIRestrictor<D>
37 {
38  private:
39  void
40  restrictPatches(const std::vector<std::pair<int, std::shared_ptr<const PatchInfo<D>>>> &patches,
41  std::shared_ptr<const Vector<D>> finer_vector,
42  std::shared_ptr<Vector<D>> coarser_vector) const override
43  {
44  for (const auto &pair : patches) {
45  std::shared_ptr<const PatchInfo<D>> pinfo = pair.second;
46  LocalData<D> coarse_local_data = coarser_vector->getLocalData(pair.first);
47  LocalData<D> fine_data = finer_vector->getLocalData(pinfo->local_index);
48 
49  if (pinfo->hasCoarseParent()) {
50  Orthant<D> orth = pinfo->orth_on_parent;
51  std::array<int, D> starts;
52  for (size_t i = 0; i < D; i++) {
53  starts[i]
54  = orth.isOnSide(Side<D>(2 * i)) ? 0 : coarse_local_data.getLengths()[i];
55  }
56 
57  nested_loop<D>(fine_data.getStart(), fine_data.getEnd(),
58  [&](const std::array<int, D> &coord) {
59  std::array<int, D> coarse_coord;
60  for (size_t x = 0; x < D; x++) {
61  coarse_coord[x] = (coord[x] + starts[x]) / 2;
62  }
63  coarse_local_data[coarse_coord] += fine_data[coord] / (1 << D);
64  });
65  } else {
66  nested_loop<D>(fine_data.getStart(), fine_data.getEnd(),
67  [&](const std::array<int, D> &coord) {
68  coarse_local_data[coord] += fine_data[coord];
69  });
70  }
71  }
72  }
73 
74  public:
80  AvgRstr(std::shared_ptr<InterLevelComm<D>> ilc_in) : MPIRestrictor<D>(ilc_in) {}
81 };
