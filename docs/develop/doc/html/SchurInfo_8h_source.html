112 template <size_t D> class NormalIfaceInfo : public IfaceInfo<D>
113 {
114  public:
118  std::shared_ptr<NormalNbrInfo<D>> nbr_info;
122  NormalIfaceInfo()
123  {
124  this->id = 0;
125  this->local_index = 0;
126  this->global_index = 0;
127  }
134  NormalIfaceInfo(std::shared_ptr<PatchInfo<D>> pinfo, Side<D> s)
135  {
136  nbr_info = pinfo->getNormalNbrInfoPtr(s);
137  if (s.isLowerOnAxis()) {
138  this->id = pinfo->id * Side<D>::num_sides + s.getIndex();
139  // lower axis interface belongs to neighboring rank
140  this->rank = nbr_info->rank;
141  } else {
142  this->id = nbr_info->id * Side<D>::num_sides + s.opposite().getIndex();
143  // higher axis interafce belongs to this patch's rank
144  this->rank = pinfo->rank;
145  }
146  }
147  void getIds(std::deque<int> &ids)
148  {
149  ids.push_back(this->id);
150  }
151  void getLocalIndexes(std::deque<int> &idx)
152  {
153  idx.push_back(this->local_index);
154  }
155  void getGlobalIndexes(std::deque<int> &idx)
156  {
157  idx.push_back(this->global_index);
158  }
159  void getIfaceTypes(std::deque<IfaceType<D>> &types)
160  {
161  types.push_back(IfaceType<D>::normal);
162  }
163  void getRanks(std::deque<int> &ranks)
164  {
165  ranks.push_back(nbr_info->rank);
166  }
167  void setLocalIndexes(const std::map<int, int> &rev_map)
168  {
169  this->local_index = rev_map.at(this->id);
170  }
171  void setGlobalIndexes(const std::map<int, int> &rev_map)
172  {
173  this->global_index = rev_map.at(this->local_index);
174  }
175 };
184 template <size_t D> class CoarseIfaceInfo : public IfaceInfo<D>
185 {
186  public:
190  std::shared_ptr<CoarseNbrInfo<D>> nbr_info;
194  Orthant<D - 1> orth_on_coarse;
198  int coarse_rank;
202  int coarse_id;
206  int coarse_local_index;
210  int coarse_global_index;
217  CoarseIfaceInfo(std::shared_ptr<PatchInfo<D>> pinfo, Side<D> s)
218  {
219  nbr_info = pinfo->getCoarseNbrInfoPtr(s);
220  this->id = pinfo->id * Side<D>::num_sides + s.getIndex();
221  orth_on_coarse = nbr_info->orth_on_coarse;
222  coarse_id = nbr_info->id * Side<D>::num_sides + s.opposite().getIndex();
223  // fine and coarse interfaces always belong to their patches
224  this->rank = pinfo->rank;
225  this->coarse_rank = nbr_info->rank;
226  }
227  void getIds(std::deque<int> &ids)
228  {
229  ids.push_back(this->id);
230  ids.push_back(coarse_id);
231  }
232  void getLocalIndexes(std::deque<int> &idx)
233  {
234  idx.push_back(this->local_index);
235  idx.push_back(coarse_local_index);
236  }
237  void getGlobalIndexes(std::deque<int> &idx)
238  {
239  idx.push_back(this->global_index);
240  idx.push_back(coarse_global_index);
241  }
242  void getIfaceTypes(std::deque<IfaceType<D>> &types)
243  {
244  IfaceType<D> fine_type(IfaceType<D>::fine_to_fine, orth_on_coarse);
245  IfaceType<D> coarse_type(IfaceType<D>::fine_to_coarse, orth_on_coarse);
246  types.push_back(fine_type);
247  types.push_back(coarse_type);
248  }
249  void getRanks(std::deque<int> &ranks)
250  {
251  ranks.push_back(nbr_info->rank);
252  ranks.push_back(nbr_info->rank);
253  }
254  void setLocalIndexes(const std::map<int, int> &rev_map)
255  {
256  this->local_index = rev_map.at(this->id);
257  auto it = rev_map.find(coarse_id);
258  if (it != rev_map.end())
259  coarse_local_index = it->second;
260  }
261  void setGlobalIndexes(const std::map<int, int> &rev_map)
262  {
263  this->global_index = rev_map.at(this->local_index);
264  coarse_global_index = rev_map.at(coarse_local_index);
265  }

275 template <size_t D> class FineIfaceInfo : public IfaceInfo<D>
276 {
277  public:
281  std::shared_ptr<FineNbrInfo<D>> nbr_info;
285  std::array<int, Orthant<D - 1>::num_orthants> fine_ranks;
289  std::array<int, Orthant<D - 1>::num_orthants> fine_ids;
293  std::array<int, Orthant<D - 1>::num_orthants> fine_local_indexes;
297  std::array<int, Orthant<D - 1>::num_orthants> fine_global_indexes;
304  FineIfaceInfo(std::shared_ptr<PatchInfo<D>> pinfo, Side<D> s)
305  {
306  nbr_info = pinfo->getFineNbrInfoPtr(s);
307  this->id = pinfo->id * Side<D>::num_sides + s.getIndex();
308  this->rank = pinfo->rank;
309  for (size_t i = 0; i < fine_ids.size(); i++) {
310  fine_ids[i] = nbr_info->ids[i] * Side<D>::num_sides + s.opposite().getIndex();
311  fine_ranks[i] = nbr_info->ranks[i];
312  }
313  }
314  void getIdxAndTypes(std::deque<int> &idx, std::deque<IfaceType<D>> &types)
315  {
316  idx.push_back(this->local_index);
317  types.push_back(IfaceType<D>::coarse_to_coarse);
318  for (size_t i = 0; i < fine_local_indexes.size(); i++) {
319  idx.push_back(fine_local_indexes[i]);
320  IfaceType<D> type(IfaceType<D>::coarse_to_fine, i);
321  types.push_back(type);
322  }
323  }
324  void getIds(std::deque<int> &ids)
325  {
326  ids.push_back(this->id);
327  for (size_t i = 0; i < fine_ids.size(); i++) {
328  ids.push_back(fine_ids[i]);
329  }
330  }
331  void getLocalIndexes(std::deque<int> &idx)
332  {
333  idx.push_back(this->local_index);
334  for (size_t i = 0; i < fine_ids.size(); i++) {
335  idx.push_back(fine_local_indexes[i]);
336  }
337  }
338  void getGlobalIndexes(std::deque<int> &idx)
339  {
340  idx.push_back(this->global_index);
341  for (size_t i = 0; i < fine_ids.size(); i++) {
342  idx.push_back(fine_global_indexes[i]);
343  }
344  }
345  void getIfaceTypes(std::deque<IfaceType<D>> &types)
346  {
347  types.push_back(IfaceType<D>::coarse_to_coarse);
348  for (Orthant<D - 1> o : Orthant<D - 1>::getValues()) {
349  IfaceType<D> type(IfaceType<D>::coarse_to_fine, o);
350  types.push_back(type);
351  }
352  }
353  void getRanks(std::deque<int> &ranks)
354  {
355  ranks.push_back(-1);
356  for (size_t i = 0; i < fine_ids.size(); i++) {
357  ranks.push_back(nbr_info->ranks[i]);
358  }
359  }
360  void setLocalIndexes(const std::map<int, int> &rev_map)
361  {
362  this->local_index = rev_map.at(this->id);
363  for (size_t i = 0; i < fine_ids.size(); i++) {
364  auto it = rev_map.find(this->fine_ids[i]);
365  if (it != rev_map.end())
366  fine_local_indexes[i] = it->second;
367  }
368  }
369  void setGlobalIndexes(const std::map<int, int> &rev_map)
370  {
371  this->global_index = rev_map.at(this->local_index);
372  for (size_t i = 0; i < fine_local_indexes.size(); i++) {
373  fine_global_indexes[i] = rev_map.at(fine_local_indexes[i]);
374  }
375  }
376 };
382 template <size_t D> struct SchurInfo : public Serializable {
386  std::shared_ptr<PatchInfo<D>> pinfo;
390  std::array<std::shared_ptr<IfaceInfo<D>>, Side<D>::num_sides> iface_info;
395  SchurInfo()
396  {
397  iface_info.fill(nullptr);
398  }
404  SchurInfo(std::shared_ptr<PatchInfo<D>> &pinfo)
405  {
406  this->pinfo = pinfo;
407  iface_info.fill(nullptr);
408 
409  // create iface objects
410  for (Side<D> s : Side<D>::getValues()) {
411  if (pinfo->hasNbr(s)) {
412  switch (pinfo->getNbrType(s)) {
413  case NbrType::Normal:
414  getIfaceInfoPtr(s).reset(new NormalIfaceInfo<D>(pinfo, s));
415  break;
416  case NbrType::Fine:
417  getIfaceInfoPtr(s).reset(new FineIfaceInfo<D>(pinfo, s));
418  break;
419  case NbrType::Coarse:
420  getIfaceInfoPtr(s).reset(new CoarseIfaceInfo<D>(pinfo, s));
421  break;
422  }
423  }
424  }
425  }
431  std::shared_ptr<IfaceInfo<D>> &getIfaceInfoPtr(Side<D> s)
432  {
433  return iface_info[s.getIndex()];
434  }
435  /*
436  * @brief Get a reference to the NormalIfaceInfo object.
437  *
438  * If there is not a NormalIfaceInfo object on this side, the behavior is undefined.
439  *
440  * @param s the side of the patch that the interface is on
441  */
442  NormalIfaceInfo<D> &getNormalIfaceInfo(Side<D> s)
443  {
444  return *std::dynamic_pointer_cast<NormalIfaceInfo<D>>(iface_info[s.getIndex()]);
445  }
446  /*
447  * @brief Get a reference to the CoarseIfaceInfo object.
448  *
449  * If there is not a CoarseIfaceInfo object on this side, the behavior is undefined.
450  *
451  * @param s the side of the patch that the interface is on
452  */
453  CoarseIfaceInfo<D> &getCoarseIfaceInfo(Side<D> s)
454  {
455  return *std::dynamic_pointer_cast<CoarseIfaceInfo<D>>(iface_info[s.getIndex()]);
456  }
457  /*
458  * @brief Get a reference to the FineIfaceInfo object.
459  *
460  * If there is not a FineIfaceInfo object on this side, the behavior is undefined.
461  *
462  * @param s the side of the patch that the interface is on
463  */
464  FineIfaceInfo<D> &getFineIfaceInfo(Side<D> s)
465  {
466  return *std::dynamic_pointer_cast<FineIfaceInfo<D>>(iface_info[s.getIndex()]);
467  }
468  /*
469  * @brief Get a reference to the NormalIfaceInfo object.
470  *
471  * If there is not a NormalIfaceInfo object on this side, the behavior is undefined.
472  *
473  * @param s the side of the patch that the interface is on
474  */
475  const NormalIfaceInfo<D> &getNormalIfaceInfo(Side<D> s) const
476  {
477  return *std::dynamic_pointer_cast<NormalIfaceInfo<D>>(iface_info[s.getIndex()]);
478  }
479  /*
480  * @brief Get a reference to the CoarseIfaceInfo object.
481  *
482  * If there is not a CoarseIfaceInfo object on this side, the behavior is undefined.
483  *
484  * @param s the side of the patch that the interface is on
485  */
486  const CoarseIfaceInfo<D> &getCoarseIfaceInfo(Side<D> s) const
487  {
488  return *std::dynamic_pointer_cast<CoarseIfaceInfo<D>>(iface_info[s.getIndex()]);
489  }
490  /*
491  * @brief Get a reference to the FineIfaceInfo object.
492  *
493  * If there is not a FineIfaceInfo object on this side, the behavior is undefined.
494  *
495  * @param s the side of the patch that the interface is on
496  */
497  const FineIfaceInfo<D> &getFineIfaceInfo(Side<D> s) const
498  {
499  return *std::dynamic_pointer_cast<FineIfaceInfo<D>>(iface_info[s.getIndex()]);
500  }
509  std::deque<int> getIds()
510  {
511  std::deque<int> retval;
512  for (Side<D> s : Side<D>::getValues()) {
513  if (pinfo->hasNbr(s)) {
514  getIfaceInfoPtr(s)->getIds(retval);
515  }
516  }
517  return retval;
518  }
519  void setLocalIndexes(const std::map<int, int> &rev_map)
520  {
521  for (Side<D> s : Side<D>::getValues()) {
522  if (pinfo->hasNbr(s)) {
523  getIfaceInfoPtr(s)->setLocalIndexes(rev_map);
524  }
525  }
526  }
527  void setGlobalIndexes(const std::map<int, int> &rev_map)
528  {
529  for (Side<D> s : Side<D>::getValues()) {
530  if (pinfo->hasNbr(s)) {
531  getIfaceInfoPtr(s)->setGlobalIndexes(rev_map);
532  }
533  }
534  }
535  int getIfaceLocalIndex(Side<D> s) const
536  {
537  return iface_info[s.getIndex()]->local_index;
538  }
539  int serialize(char *buffer) const
540  {
541  BufferWriter writer(buffer);
542  writer << *pinfo;
543  return writer.getPos();
544  }
545  int deserialize(char *buffer)
546  {
547  BufferReader reader(buffer);
548  pinfo.reset(new PatchInfo<D>());
549  reader >> *pinfo;
550  // create iface objects
551  for (Side<D> s : Side<D>::getValues()) {
552  if (pinfo->hasNbr(s)) {
553  switch (pinfo->getNbrType(s)) {
554  case NbrType::Normal:
555  getIfaceInfoPtr(s).reset(new NormalIfaceInfo<D>(pinfo, s));
556  break;
557  case NbrType::Fine:
558  getIfaceInfoPtr(s).reset(new FineIfaceInfo<D>(pinfo, s));
559  break;
560  case NbrType::Coarse:
561  getIfaceInfoPtr(s).reset(new CoarseIfaceInfo<D>(pinfo, s));
562  break;
563  }
564  }
565  }
566  return reader.getPos();
567  }
