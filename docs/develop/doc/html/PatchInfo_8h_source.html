305  void setNeumann(IsNeumannFunc<D> inf)
306  {
307  for (Side<D> s : Side<D>::getValues()) {
308  if (!hasNbr(s)) {
309  std::array<double, D> bound_start = starts;
310  if (!s.isLowerOnAxis()) {
311  bound_start[s.getAxisIndex()]
312  += spacings[s.getAxisIndex()] * ns[s.getAxisIndex()];
313  }
314  std::array<double, D> bound_end = bound_start;
315  for (size_t dir = 0; dir < D; dir++) {
316  if (dir != s.getAxisIndex()) {
317  bound_end[dir] += spacings[dir] * ns[dir];
318  }
319  }
320  neumann[s.getIndex()] = inf(s, bound_end, bound_start);
321  }
322  }

390  int serialize(char *buffer) const
391  {
392  BufferWriter writer(buffer);
393  writer << id;
394  writer << ns;
395  writer << refine_level;
396  writer << parent_id;
397  writer << parent_rank;
398  writer << child_ids;
399  writer << child_ranks;
400  writer << rank;
401  writer << orth_on_parent;
402  writer << neumann;
403  writer << starts;
404  writer << spacings;
405  std::bitset<Side<D>::num_sides> has_nbr;
406  for (size_t i = 0; i < Side<D>::num_sides; i++) {
407  has_nbr[i] = nbr_info[i] != nullptr;
408  }
409  writer << has_nbr;
410  for (Side<D> s : Side<D>::getValues()) {
411  if (hasNbr(s)) {
412  NbrType type = getNbrType(s);
413  writer << type;
414  switch (type) {
415  case NbrType::Normal: {
416  NormalNbrInfo<D> tmp = getNormalNbrInfo(s);
417  writer << tmp;
418  } break;
419  case NbrType::Fine: {
420  FineNbrInfo<D> tmp = getFineNbrInfo(s);
421  writer << tmp;
422  } break;
423  case NbrType::Coarse: {
424  CoarseNbrInfo<D> tmp = getCoarseNbrInfo(s);
425  writer << tmp;
426  } break;
427  }
428  }
429  }
430  return writer.getPos();
431  }
432  int deserialize(char *buffer)
433  {
434  BufferReader reader(buffer);
435  reader >> id;

437  reader >> refine_level;
438  reader >> parent_id;
439  reader >> parent_rank;
440  reader >> child_ids;
441  reader >> child_ranks;
442  reader >> rank;
443  reader >> orth_on_parent;
444  reader >> neumann;
445  reader >> starts;
446  reader >> spacings;
447  std::bitset<Side<D>::num_sides> has_nbr;
448  reader >> has_nbr;
449  for (size_t i = 0; i < Side<D>::num_sides; i++) {
450  if (has_nbr[i]) {
451  NbrType type;
452  reader >> type;
453  std::shared_ptr<NbrInfo<D>> info;
454  switch (type) {
455  case NbrType::Normal:
456  info.reset(new NormalNbrInfo<D>());
457  reader >> *std::static_pointer_cast<NormalNbrInfo<D>>(info);
458  break;
459  case NbrType::Fine:
460  info.reset(new FineNbrInfo<D>());
461  reader >> *std::static_pointer_cast<FineNbrInfo<D>>(info);
462  break;
463  case NbrType::Coarse:
464  info.reset(new CoarseNbrInfo<D>());
465  reader >> *std::static_pointer_cast<CoarseNbrInfo<D>>(info);
466  break;
467  }
468  nbr_info[i] = info;
469  }
470  }
471  return reader.getPos();
472  }

474 template <int D> void to_json(nlohmann::json &j, const PatchInfo<D> &pinfo)
475 {
476  j["id"] = pinfo.id;
477  j["parent_id"] = pinfo.parent_id;
478  j["parent_rank"] = pinfo.parent_rank;
479  j["orth_on_parent"] = pinfo.orth_on_parent;
480  j["rank"] = pinfo.rank;
481  j["starts"] = pinfo.starts;
482  j["lengths"] = pinfo.spacings;
483  for (int i = 0; i < D; i++) {
484  j["lengths"][i] = pinfo.spacings[i] * pinfo.ns[i];
485  }
486  j["nbrs"] = nlohmann::json::array();
487  for (Side<D> s : Side<D>::getValues()) {
488  if (pinfo.hasNbr(s)) {
489  switch (pinfo.getNbrType(s)) {
490  case NbrType::Normal:
491  j["nbrs"].push_back(pinfo.getNormalNbrInfo(s));
492  break;
493  case NbrType::Coarse:
494  j["nbrs"].push_back(pinfo.getCoarseNbrInfo(s));
495  break;
496  case NbrType::Fine:
497  j["nbrs"].push_back(pinfo.getFineNbrInfo(s));
498  break;
499  default:
500  throw RuntimeError("Unsupported NbrType");
501  }
502  j["nbrs"].back()["type"] = pinfo.getNbrType(s);
503  j["nbrs"].back()["side"] = s;
504  }
505  }
506  if (pinfo.child_ids[0] != -1) {
507  j["child_ids"] = pinfo.child_ids;
508  j["child_ranks"] = pinfo.child_ranks;
509  }
510 }
511 template <int D> void from_json(const nlohmann::json &j, PatchInfo<D> &pinfo)
512 {
513  pinfo.id = j["id"];
514  pinfo.parent_id = j["parent_id"];
515  pinfo.parent_rank = j["parent_rank"];
516  if (j.contains("orth_on_parent")) {
517  j["orth_on_parent"].get_to(pinfo.orth_on_parent);
518  }
519  pinfo.rank = j["rank"];
520  pinfo.starts = j["starts"].get<std::array<double, D>>();
521  pinfo.spacings = j["lengths"].get<std::array<double, D>>();
522  pinfo.ns.fill(1);
523  for (const auto &nbr_j : j["nbrs"]) {
524  Side<D> s = nbr_j["side"].get<Side<D>>();
525  switch (nbr_j["type"].get<NbrType>()) {
526  case NbrType::Normal:
527  pinfo.nbr_info[s.getIndex()] = std::make_shared<NormalNbrInfo<D>>();
528  pinfo.getNormalNbrInfo(s) = nbr_j.get<NormalNbrInfo<D>>();
529  break;
530  case NbrType::Coarse:
531  pinfo.nbr_info[s.getIndex()] = std::make_shared<CoarseNbrInfo<D>>();
532  pinfo.getCoarseNbrInfo(s) = nbr_j.get<CoarseNbrInfo<D>>();
533  break;
534  case NbrType::Fine:
535  pinfo.nbr_info[s.getIndex()] = std::make_shared<FineNbrInfo<D>>();
536  pinfo.getFineNbrInfo(s) = nbr_j.get<FineNbrInfo<D>>();
537  break;
538  default:
539  throw RuntimeError("Unsupported NbrType");
540  }
541  }
542  if (j.contains("child_ids")) {
543  pinfo.child_ids = j["child_ids"].get<std::array<int, Orthant<D>::num_orthants>>();
544  pinfo.child_ranks = j["child_ranks"].get<std::array<int, Orthant<D>::num_orthants>>();
545  }
