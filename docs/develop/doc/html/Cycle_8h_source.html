54  void prepCoarser(const Level<D> &level, VecList &u_vectors, ConstVecList &f_vectors) const
55  {
56  // calculate residual
57  std::shared_ptr<Vector<D>> r = level.getVectorGenerator()->getNewVector();
58  level.getOperator()->apply(u_vectors.front(), r);
59  r->scaleThenAdd(-1, f_vectors.front());
60  // create vectors for coarser levels
61  std::shared_ptr<Vector<D>> new_u = level.getCoarser()->getVectorGenerator()->getNewVector();
62  std::shared_ptr<Vector<D>> new_f = level.getCoarser()->getVectorGenerator()->getNewVector();
63  level.getRestrictor()->restrict(r, new_f);
64  u_vectors.push_front(new_u);
65  f_vectors.push_front(new_f);
66  }
72  void prepFiner(const Level<D> &level, VecList &u_vectors, ConstVecList &f_vectors) const
73  {
74  std::shared_ptr<Vector<D>> old_u = u_vectors.front();
75  u_vectors.pop_front();
76  f_vectors.pop_front();
77  level.getInterpolator()->interpolate(old_u, u_vectors.front());
78  }
79 
85  void smooth(const Level<D> &level, VecList &u_vectors, ConstVecList &f_vectors) const
86  {
87  level.getSmoother()->smooth(f_vectors.front(), u_vectors.front());
88  }

104  Cycle(std::shared_ptr<Level<D>> finest_level)
105  {

114  void apply(std::shared_ptr<const Vector<D>> f, std::shared_ptr<Vector<D>> u) const
115  {
116  u->set(0);
117  VecList u_vectors;
118  ConstVecList f_vectors;
119  f_vectors.push_back(f); /* Appends */
120  u_vectors.push_back(u);
121  visit(*finest_level, u_vectors, f_vectors);
122  f_vectors.pop_front();
123  u_vectors.pop_front();
124  }
130  std::shared_ptr<const Level<D>> getFinestLevel() const

133  }
